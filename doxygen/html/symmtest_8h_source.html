105 void TestMatrix(const smatd& matr, double tol) {
106 TTypeList< AnAssociation3<vec3d,size_t,size_t> > tmpVecs;
107 Vecs.Clear();
108 Vecs.SetCapacity( (Atoms.Count()*Atoms.Count()-1)/2+1 );
109 tmpVecs.SetCapacity( (Atoms.Count()*Atoms.Count()-1)/2+1 );
110 for( size_t i=0; i < Atoms.Count(); i++ ) {
111 for( size_t j=i; j < Atoms.Count(); j++ ) {
112 vec3d a = matr * Atoms[j].GetA() - Atoms[i].GetA();
113 for( int k=0; k < 3; k++ ) {
114 a[k] -= olx_round(a[k]);
115 if( a[k] < 0 ) a[k] += 1;
116 }
117 tmpVecs.AddNew<vec3d,size_t,size_t>(a, i, j);
118 }
119 }
120
121 // get the square, as we compare quadratic distances
122 const double stol = 3*tol*tol;
123
124 // optimisation technique
125 QuickSorter::SortSF(tmpVecs,VecsCmpByRadius);
126 float* radii = new float[ tmpVecs.Count()+1];
127 for( size_t i=0; i < tmpVecs.Count(); i++ )
128 radii[i] = (float)tmpVecs[i].GetA().QLength();
129
130 // collect the results
131 for( size_t i=0; i < tmpVecs.Count(); i++ ) {
132 if( tmpVecs.IsNull(i) ) continue;
133 TSymmTestData* sd = new TSymmTestData(
134 tmpVecs[i].GetA(), tmpVecs[i].GetB(), tmpVecs[i].GetC());
135 Vecs.Add(*sd);
136 double qval = radii[i];
137 tmpVecs.NullItem(i);
138 for( size_t j=i+1; j < tmpVecs.Count(); j++ ) {
139 if( (radii[j] - qval) > stol ) break;
140 if( tmpVecs.IsNull(j) ) continue;
141 sd->Atoms.AddNew<size_t,size_t>( tmpVecs[j].GetB(), tmpVecs[j].GetC() );
142 qval *= 0.75;
143 qval += radii[j]*0.25;
144 sd->Center += tmpVecs[j].GetA(); // accumulating center
145 tmpVecs.NullItem(j);
146 }
147 }
148 // do the averaging
149 for( size_t i=0; i < Vecs.Count(); i++ )
150 Vecs[i].Center /= Vecs[i].Count();
151
152 delete [] radii;
153 QuickSorter::SortSF(Vecs,VecsCmpByCount);
154 }
155 static void TestDependency(
156 const TTypeList< olx_pair_t<vec3d,TCAtom*> >& lista,
157 const TTypeList< olx_pair_t<vec3d,TCAtom*> >& listb,
158 TTypeList< TSymmTestData >& Vecs,
159 const smatd& matr, double tol)
160 {
161 TTypeList< AnAssociation3<vec3d,size_t,size_t> > tmpVecs;
162 Vecs.Clear();
163 Vecs.SetCapacity( lista.Count()*listb.Count()+1 );
164 tmpVecs.SetCapacity( lista.Count()*listb.Count()+1 );
165 vec3d a, b;
166 for( size_t i=0; i < lista.Count(); i++ ) {
167 for( size_t j=0; j < listb.Count(); j++ ) {
168 a = lista[i].GetA();
169 b = listb[j].GetA();
170 b = matr * b;
171 a += b;
172 for( int k=0; k < 3; k++ ) {
173 a[k] -= olx_round(a[k]);
174 if( a[k] < 0 ) a[k] += 1;
175 if( a[k] < tol ) a[k] = 0;
176 else if( (1-a[k]) < tol ) a[k] = 0;
177 }
178 tmpVecs.AddNew<vec3d,size_t,size_t>(a, i, j);
179 }
180 }
181
182 // get the square, as we compare quadratic distances
183 double stol = 3*tol*tol, qval;
184
185 // optimisation technique
186 QuickSorter::SortSF(tmpVecs,VecsCmpByRadius);
187 float* radii = new float[ tmpVecs.Count()+1];
188 for( size_t i=0; i < tmpVecs.Count(); i++ )
189 radii[i] = (float)tmpVecs[i].GetA().QLength();
190
191 // collect the results
192 for( size_t i=0; i < tmpVecs.Count(); i++ ) {
193 if( tmpVecs.IsNull(i) ) continue;
194 TSymmTestData* sd = new TSymmTestData(
195 tmpVecs[i].GetA(), tmpVecs[i].GetB(), tmpVecs[i].GetC());
196 Vecs.Add(*sd);
197 a = tmpVecs[i].GetA(); // "crawling" center
198 qval = radii[i];
199 tmpVecs.NullItem(i);
200 for( size_t j=i+1; j < tmpVecs.Count(); j++ ) {
201 if( (radii[j] - qval) > stol ) break;
202 if( tmpVecs.IsNull(j) ) continue;
203 if( olx_abs(tmpVecs[j].GetA()[0]-a[0]) < tol &&
204 olx_abs(tmpVecs[j].GetA()[1]-a[1]) < tol &&
205 olx_abs(tmpVecs[j].GetA()[2]-a[2]) < tol ) {
206 sd->Atoms.AddNew<size_t,size_t>(tmpVecs[j].GetB(), tmpVecs[j].GetC());
207 a *= 0.75;
208 a += tmpVecs[j].GetA()*0.25; // implement "crawling"
209

211 qval += radii[j]*0.25;
212 //qval /= 2;
