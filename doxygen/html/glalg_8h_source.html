25 /* v1, v2 - define the vector, N(ormal) and p (point on the plane)
26 */
27 static bool VectorPlaneIntersection(const vec_t& v1, const vec_t& v2,
28 const vec_t& N, const vec_t& p, vec_t &out)
29 {
30 vec_t vd = v2 - v1;
31 float_type dp = N.DotProd(vd);
32 if (dp == 0) {
33 return false;
34 }
35 float_type u = N.DotProd(p - v1) / dp;
36 out = v1 + vd * u;
37 return true;
38 }
39 // for zero based vector
40 static bool VectorPlaneIntersection(const vec_t& v,
41 const vec_t& N, const vec_t& p, vec_t &out)
42 {
43 float_type dp = N.DotProd(v);
44 if (dp == 0) {
45 return false;
46 }
47 float_type u = N.DotProd(p) / dp;
48 out = v * u;
49 return true;
50 }
51 /*
52 checks if the line segment intersects with the given plane
53 */
54 static bool DoLineAndPlaneIntersect(const vec_t& v1, const vec_t& v2,
55 const vec_t& N, const vec_t& p)
56 {
57 vec_t vd = v2 - v1;
58 float_type dp = N.DotProd(vd);
59 if (dp == 0) {
60 return false;
61 }
62 float_type u = N.DotProd(p - v1) / dp;
63 return u >= 0 && u <= 1;
64 }

66 static bool IsPointInTriangle(const vec_t& p,
67 const vec_t& a, const vec_t& b, const vec_t& c)
68 {
69 // a-b-c
70 if (((a - b).XProdVec(p - b)).DotProd((c - b).XProdVec(p - b)) > 0)
71 return false;
72 // c-a-b
73 if (((c - a).XProdVec(p - a)).DotProd((b - a).XProdVec(p - a)) > 0)
74 return false;
75 // b-c-a
76 if (((b - c).XProdVec(p - c)).DotProd((a - c).XProdVec(p - c)) > 0)
77 return false;
78 return true;
79 }

81 /*
82 ao - line a origin, ad - line a direction
83 bo - line b origin, bd - line b direction
84 output: pa - point on line a, pb - point on line b defining the shortest
85 segment
86 */
87 static bool LineLineIntersect(const vec_t &ao, const vec_t& ad,
88 const vec_t& bo, const vec_t& bd,
89 vec_t &pa, vec_t &pb)
90 {
91 vec_t ab = ao - bo;
92 float_type d1343 = ab.DotProd(bd);
93 float_type d4321 = bd.DotProd(ad);
94 float_type d1321 = ab.DotProd(ad);
95 float_type d2121 = ad.QLength();
96 float_type d4343 = bd.QLength();
97 float_type denom = d2121 * d4343 - d4321 * d4321;
98 if (olx_abs(denom) < 1e-6) {
99 return false;
100 }
101 float_type numer = d1343 * d4321 - d1321 * d4343;
102 float_type mua = numer / denom;
103 float_type mub = (d1343 + d4321 * mua) / d4343;
104 pa = ao + ad * mua;
105 pb = bo + bd * mub;
106 return true;
107 }
108 /*
109 ao - a point on plane a, an - normal for plane a,
110 bo - a point on plane b, an - normal for plane b,
111 output: lo - a point on the intersection line, ld - direction
112 */
113 static bool PlanePlaneIntersect(const vec_t &ao, const vec_t &an,
114 const vec_t &bo, const vec_t &bn, vec_t &lo, vec_t &ld)
115 {
116 float_type np = an.DotProd(bn);
117 if (np == 0) {
118 return false;
119 }
120 float_type an_ql = an.QLength(),
121 bn_ql = bn.QLength(),
122 adn = an.DotProd(ao),
123 bdn = bn.DotProd(bo);
124 float_type dv = an_ql * bn_ql - olx_sqr(np);
125 float_type c1 = (adn * bn_ql - bdn * np) / dv,
126 c2 = (bdn * an_ql - adn * np) / dv;
127 lo = an * c1 + bn * c2;
128 ld = an.XProdVec(bn);
129 return true;
130 }

132 struct DiffTask {
133 const TTypeList<vec_t>& v1, & v2;
134 const TTypeList<IndexTriangle>& triags;
135 const TArrayList<float_type>& QL;
136 TEBitArray& used, & used1;
137 typedef gl_alg<float_type> alg;
138 DiffTask(const TTypeList<vec_t>& v1, const TTypeList<vec_t>& v2,
139 const TTypeList<IndexTriangle>& triags,
140 const TArrayList<float_type>& QL,
141 TEBitArray& used,
142 TEBitArray& used1)
143 : v1(v1), v2(v2), triags(triags),
144 QL(QL), used(used), used1(used1)
145 {}
146 void Run(size_t i) {
147 float_type ql = v1[i].QLength();
148 for (size_t j = 0; j < triags.Count(); j++) {
149 const IndexTriangle& t = triags[j];
150 if (used1[j] || (ql > QL[t[0]] && ql > QL[t[1]] && ql > QL[t[2]]))
151 continue;
152 vec_t p = alg::VectorPlaneIntersection(v1[i],
153 (v2[t[0]] - v2[t[1]]).XProdVec(v2[t[2]] - v2[t[1]]).Normalise(),
154 v2[t[2]]);
155 if (alg::IsPointInTriangle(p, v2[t[0]], v2[t[1]], v2[t[2]])) {
156 used.SetTrue(i);
157 used1.SetTrue(j);
158 }
159 }
160 }
161 DiffTask* Replicate() const {
162 return new DiffTask(v1, v2, triags, QL, used, used1);
163 }
164 };

174 TTypeList<IndexTriangle>& out2)
175 {
176 TArrayList<float_type> QL(v2.Count());
177 TEBitArray used(v1.Count()), used1(triags.Count());
178 for (size_t i = 0; i < v2.Count(); i++)
179 QL[i] = v2[i].QLength();
180 DiffTask dt(v1, v2, triags, QL, used, used1);
181 OlxListTask::Run(dt, v1.Count(), tLinearTask, 1000);
182 out1.SetCapacity(out1.Count() + triags.Count());
183 out2.SetCapacity(out2.Count() + triags.Count());
184 for (size_t i = 0; i < triags.Count(); i++) {
185 const IndexTriangle& t = triags[i];
186 if (used[t[0]] || used[t[1]] || used[t[2]]) {
187 out1.AddCopy(t);
188 }
189 if (used1[t[0]] || used1[t[1]] || used1[t[2]]) {
190 out2.AddCopy(t);
191 }
192 }
193 }
