15struct HashingUtils {
16 static inline uint32_t hs_rotl(uint32_t v, uint32_t c) {
17 return (v << c) | (v >> (32 - c));
18 }
19 static inline uint32_t hs_rotr(uint32_t v, uint32_t c) {
20 return (v >> c) | (v << (32 - c));
21 }
22};
23struct HashingUtilsLE : public HashingUtils {
24 // operates on 64 byte blocks only, little endian
25 static inline void hs_copy(const uint8_t* src, uint32_t* dest, size_t src_len) {
26 for (size_t i = 0, j = 0; i < src_len; i += 4, j++)
27 dest[j] = ((uint32_t)src[i]) |
28 (((uint32_t)src[i + 1]) << 8) |
29 (((uint32_t)src[i + 2]) << 16) |
30 (((uint32_t)src[i + 3]) << 24);
31 }
32 static inline void hs_copy(const uint32_t* src, uint8_t* dest, size_t src_len) {
33 for (size_t i = 0, j = 0; j < src_len; i += 4, j++) {
34 dest[i] = (char)src[j];
35 dest[i + 1] = (char)(src[j] >> 8);
36 dest[i + 2] = (char)(src[j] >> 16);
37 dest[i + 3] = (char)(src[j] >> 24);
38 }
39 }
40 static inline void hs_write_len(uint8_t* dest, const uint64_t& len) {
41 for (int i = 8; i >= 1; i--)
42 dest[64 - i] = ((uint8_t*)&len)[8 - i];
43 }
44};

70template <class Impl, class Tools>
71class HashingBase : public Impl {
72protected:
73 uint32_t bf[16];
74 uint8_t cbf[64];
75 void DoRawDigest(const void* msg, size_t len) {
76 const size_t blocks = len>>6;
77 for (size_t i=0; i < blocks; i++) {
78 Tools::hs_copy(&((const uint8_t*)msg)[i<<6], bf, 64);
79 Impl::digest64(bf);
80 }
81 const size_t part = len&0x3F;
82 memset(cbf, 0, 64);
83 if (part > 0) {
84 memcpy(cbf, (const void *)&(((const uint8_t *)msg)[blocks<<6]), part);
85 }
86 cbf[part] |= (0x01 << 7);
87 if (part > 56) { // will not the length fit?
88 Tools::hs_copy(cbf, bf, 64);
89 Impl::digest64(bf);
90 memset(cbf, 0, 64);
91 }
92 uint64_t len_bits_o = len<<3;
93 Tools::hs_write_len(cbf, len_bits_o);
94 Tools::hs_copy(cbf, bf, 64);
95 Impl::digest64(bf);
96 Impl::finalise();
97 }

108 for (uint64_t i=0; i < blocks; i++) {
