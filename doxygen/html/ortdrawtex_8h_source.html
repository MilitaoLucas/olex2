48 size_t PrepareArc(const TArrayList<vec3f>& in, TPtrList<const vec3f>& out, const vec3f& normal) {
49 size_t start = InvalidIndex, cnt = 0;
50 for (size_t i = 0; i < in.Count(); i++) {
51 const size_t next_i = ((i + 1) >= in.Count() ? i - in.Count() : i) + 1;
52 if (in[i].DotProd(normal) < 0 && in[next_i].DotProd(normal) >= 0) {
53 start = next_i;
54 break;
55 }
56 }
57 if (start == InvalidIndex) return 0;
58 for (size_t i = start; i < start + in.Count(); i++) {
59 const size_t ind = (i >= in.Count() ? i - in.Count() : i);
60 if (cnt + 1 >= in.Count()) break;
61 if (in[ind].DotProd(normal) >= 0)
62 out[cnt++] = &in[ind];
63 else
64 break;
65 }
66 return cnt;
67 }
68 void DoDrawBonds(TEXWriter& pw, const OrtAtom& oa, uint32_t color, float scalex, const vec3d& p) {
69 char bf[250]; //pascal
70 float xa, xb, ya, yb, za, zb, angle, xc, yc, xd, yd, xe, ye, xf, yf;
71 static int AtomCount;
72 const TSAtom& sa = *oa.atom;
73 vec3f dir_vec, touch_point, touch_point_proj, off_vec, bproj_cnt;
74 mat3f proj_mat, rot_mat;
75
76 for (size_t j = 0; j < sa.BondCount(); j++) {
77 const TXBond& bn = static_cast<TXBond&>(sa.Bond(j));
78 if (bn.IsDeleted() || !bn.IsVisible())
79 continue;
80 vec3f p1 = (bn.Another(sa).crd() + SceneOrigin) * ProjMatr + DrawOrigin;
81 if (p1[2] <= oa.crd[2]) // goes into the plane - drawn
82 continue;
83 dir_vec = (p1 - oa.crd).Normalise();
84 const float pers_scale = 1.0 - olx_sqr(dir_vec[2]);
85 float brad = (bn.A().GetType() == iHydrogenZ || bn.B().GetType() == iHydrogenZ) ?
86 BondRad * HBondScale : BondRad;
87 if (bn.GetType() == sotHBond) //even thiner H-bonds
88 brad *= HBondScale;
89 touch_point = (bn.Another(sa).crd() - oa.atom->crd()).Normalise();
90 vec3f rot_vec(-touch_point[1], touch_point[0], 0);
91 olx_create_rotation_matrix(rot_mat, rot_vec.Normalise(), touch_point[2]);
92 proj_mat = rot_mat * ProjMatr;
93 const float b_len = (p1 - oa.crd).Length();
94 if (sa.GetEllipsoid() != NULL) {
95 bproj_cnt.Null();
96 touch_point_proj = dir_vec * (*oa.ielpm);
97 for (int j = 0; j < BondDiv; j++) {
98 BondProjF[j] = (((BondCrd[j] * rot_mat + touch_point) * ProjMatr).Normalise() * (*oa.ielpm));
99 bproj_cnt += BondProjF[j];
100 }
101 bproj_cnt /= BondDiv;
102 for (int j = 0; j < BondDiv; j++) {
103 BondProjF[j] = (BondProjF[j] - bproj_cnt).NormaliseTo(brad * (1 + pers_scale) * scalex) + bproj_cnt;
104 BondProjT[j] = (BondCrd[j] * proj_mat).NormaliseTo(brad * 2 * scalex) + dir_vec * b_len;
105 }
106 }
107 else {
108 const float off_len = AradScale * sa.GetType().r_bonding / 2;
109 for (int j = 0; j < BondDiv; j++) {
110 BondProjT[j] = BondProjF[j] = (BondCrd[j] * proj_mat).NormaliseTo(brad * (1 + pers_scale) * scalex);
111 BondProjT[j].NormaliseTo(brad * 2 * scalex) += dir_vec * b_len;
112 BondProjF[j] += dir_vec * off_len;
113 }
114 }
115
116 //pw.drawQuads(BondProjF, BondProjT, &TEXWriter::fill);
117 if (BondProjF.Count() != BondProjT.Count())
118 throw TFunctionFailedException(__OlxSourceInfo, "lists mismatch");
119 if (BondProjF.Count() > 2)
120 {
121 //Calculation of the centre of the intersection ellipsis
122 za = zb = ya = yb = xa = xb = 0;
123 for (size_t bpi = 0; bpi < BondProjT.Count(); bpi++) {
124 xa += BondProjT[bpi][0];
125 ya += BondProjT[bpi][1];
126 za += BondProjT[bpi][2];
127 xb += BondProjF[bpi][0];
128 yb += BondProjF[bpi][1];
129 zb += BondProjF[bpi][2];
130 }
131 xa = xa / BondProjT.Count() + p[0];
132 xb = xb / BondProjT.Count() + p[0];
133 ya = ya / BondProjT.Count() + p[1];
134 yb = yb / BondProjT.Count() + p[1];
135 za = za / BondProjT.Count() + p[2];
136 zb = zb / BondProjT.Count() + p[2];
137 //calculation of the angle between the bond and the vertical. Needed to apply a shading on the bond
138 if (xb > xa && (xb - xa) / sqrt((xb - xa) * (xb - xa) + (yb - ya) * (yb - ya)) < 1 && (xb - xa) / sqrt((xb - xa) * (xb - xa) + (yb - ya) * (yb - ya)) > -1)
139 {
140 angle = asin((xb - xa) / sqrt((xb - xa) * (xb - xa) + (yb - ya) * (yb - ya))) * 180 / M_PI;
141 if (ya < yb)
142 angle = -asin((xb - xa) / sqrt((xb - xa) * (xb - xa) + (yb - ya) * (yb - ya))) * 180 / M_PI;
143 }
144 else if (xa > xb && (xa - xb) / sqrt((xb - xa) * (xb - xa) + (yb - ya) * (yb - ya)) < 1 && (xa - xb) / sqrt((xb - xa) * (xb - xa) + (yb - ya) * (yb - ya)) > -1)
145 {
146 angle = -asin((xa - xb) / sqrt((xb - xa) * (xb - xa) + (yb - ya) * (yb - ya))) * 180 / M_PI;
147 if (ya < yb)
148 angle = asin((xa - xb) / sqrt((xb - xa) * (xb - xa) + (yb - ya) * (yb - ya))) * 180 / M_PI;
149 }
150 else
151 {
152 angle = 0;
153 }
154
155 //To coordinate of the rectangle for the bond
156 yc = brad * (xa - xb) / sqrt(xa * xa - 2 * xa * xb + xb * xb + (ya - yb) * (ya - yb)) + ya;
157 xc = -(yb - ya) * (yc - ya) / (xb - xa) + xa;
158 ye = ya - (yc - ya);
159 xe = xa - (xc - xa);
160
161 //From coordinate of the recangle for the bond
162 //0.7, this end is a bit smaller
163 yd = 0.7 * brad * (xa - xb) / sqrt(xa * xa - 2 * xa * xb + xb * xb + (ya - yb) * (ya - yb)) + yb;
164 xd = -(ya - yb) * (yd - yb) / (xa - xb) + xb;
165 yf = yb - (yd - yb);
166 xf = xb - (xd - xb);
167
168 //white perimeter around the bond
169 sprintf(bf, "\\draw[white,very thick, rounded corners=\\cornerradius] (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- cycle;",
170 xc / 5, yc / 5,
171 xe / 5, ye / 5,
172 xf / 5, yf / 5,
173 xd / 5, yd / 5);
174 pw.Writenl(bf);
175
176 //draw the shading on the bond
177 //I need the projection coordinate of the atom
178 vec3f p2 = (bn.A().crd() + SceneOrigin) * ProjMatr + DrawOrigin;
179 vec3f p3 = (bn.B().crd() + SceneOrigin) * ProjMatr + DrawOrigin;
180
181 //All this, just to have a correct shading...
182 if (ya > yb && xa < xb)
183 {
184 if (p2[2] < p3[2])
185 sprintf(bf, "\\shade[top color=\\color%s, bottom color=\\color%s, shading angle=%f, rounded corners=\\cornerradius] (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- cycle;",
186 bn.B().GetType().symbol.c_str(), bn.A().GetType().symbol.c_str(), angle,
187 xc / 5, yc / 5,
188 xe / 5, ye / 5,
189 xf / 5, yf / 5,
190 xd / 5, yd / 5);
191 else
192 sprintf(bf, "\\shade[top color=\\color%s, bottom color=\\color%s, shading angle=%f, rounded corners=\\cornerradius] (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- cycle;",
193 bn.A().GetType().symbol.c_str(), bn.B().GetType().symbol.c_str(), angle,
194 xc / 5, yc / 5,
195 xe / 5, ye / 5,
196 xf / 5, yf / 5,
197 xd / 5, yd / 5);
198 }
199 else if (ya > yb && xa > xb)
200 {
201 if (p2[2] < p3[2])
202 sprintf(bf, "\\shade[top color=\\color%s, bottom color=\\color%s, shading angle=%f, rounded corners=\\cornerradius] (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- cycle;",
203 bn.B().GetType().symbol.c_str(), bn.A().GetType().symbol.c_str(), angle,
204 xc / 5, yc / 5,
205 xe / 5, ye / 5,
206 xf / 5, yf / 5,
207 xd / 5, yd / 5);
208 else
209 sprintf(bf, "\\shade[top color=\\color%s, bottom color=\\color%s, shading angle=%f, rounded corners=\\cornerradius] (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- cycle;",
210 bn.A().GetType().symbol.c_str(), bn.B().GetType().symbol.c_str(), angle,
211 xc / 5, yc / 5,
212 xe / 5, ye / 5,
213 xf / 5, yf / 5,
214 xd / 5, yd / 5);
215 }
216 else if (ya <= yb && xa <= xb)
217 {
218 if (p2[2] < p3[2])
219 sprintf(bf, "\\shade[top color=\\color%s, bottom color=\\color%s, shading angle=%f, rounded corners=\\cornerradius] (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- cycle;",
220 bn.A().GetType().symbol.c_str(), bn.B().GetType().symbol.c_str(), angle,
221 xc / 5, yc / 5,
222 xe / 5, ye / 5,
223 xf / 5, yf / 5,
224 xd / 5, yd / 5);
225 else
226 sprintf(bf, "\\shade[top color=\\color%s, bottom color=\\color%s, shading angle=%f, rounded corners=\\cornerradius] (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- cycle;",
227 bn.B().GetType().symbol.c_str(), bn.A().GetType().symbol.c_str(), angle,
228 xc / 5, yc / 5,
229 xe / 5, ye / 5,
230 xf / 5, yf / 5,
231 xd / 5, yd / 5);
232 }
233 else //if(ya <= yb && xa >= xb)
234 {
235 if (p2[2] < p3[2])
236 sprintf(bf, "\\shade[top color=\\color%s, bottom color=\\color%s, shading angle=%f, rounded corners=\\cornerradius] (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- cycle;",
237 bn.A().GetType().symbol.c_str(), bn.B().GetType().symbol.c_str(), angle,
238 xc / 5, yc / 5,
239 xe / 5, ye / 5,
240 xf / 5, yf / 5,
241 xd / 5, yd / 5);
242 else
243 sprintf(bf, "\\shade[top color=\\color%s, bottom color=\\color%s, shading angle=%f, rounded corners=\\cornerradius] (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- cycle;",
244 bn.B().GetType().symbol.c_str(), bn.A().GetType().symbol.c_str(), angle,
245 xc / 5, yc / 5,
246 xe / 5, ye / 5,
247 xf / 5, yf / 5,
248 xd / 5, yd / 5);
249 }
250
251 pw.Writenl(bf);
252
253 //Draw the black stroke around the bond
254 sprintf(bf, "\\draw[rounded corners=\\cornerradius] (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- (%fmm,%fmm) -- cycle;",
255 xc / 5, yc / 5,
256 xe / 5, ye / 5,
257 xf / 5, yf / 5,
258 xd / 5, yd / 5);
259 pw.Writenl(bf);
260
261 }

322 // create ellipse and pie coordinates
323 void Init(TEXWriter& pw) {
324 ElpCrd.SetCount(ElpDiv);
325 Arc.SetCount(ElpDiv);
326 PieCrd.SetCount(PieDiv);
327 FilteredArc.SetCount(ElpDiv);
328 BondCrd.SetCount(BondDiv);
329 BondProjF.SetCount(BondDiv);
330 BondProjT.SetCount(BondDiv);
331 double sin_a, cos_a;
332 olx_sincos(2 * M_PI / ElpDiv, &sin_a, &cos_a);
333 vec3f ps(cos_a, -sin_a, 0);
334 for (int i = 0; i < ElpDiv; i++) {
335 ElpCrd[i] = ps;
336 const float x = ps[0];
337 ps[0] = (float)(cos_a * x + sin_a * ps[1]);
338 ps[1] = (float)(cos_a * ps[1] - sin_a * x);
339 }
340 olx_sincos(2 * M_PI / BondDiv, &sin_a, &cos_a);
341 ps = vec3f(cos_a / 2, -sin_a / 2, 0);
342 for (int i = 0; i < BondDiv; i++) {
343 BondCrd[i] = ps;
344 const float x = ps[0];
345 ps[0] = (float)(cos_a * x + sin_a * ps[1]);
346 ps[1] = (float)(cos_a * ps[1] - sin_a * x);
347 }
348 ps = vec3f(1, 0, 0);
349 for (int i = 0; i < PieDiv; i++) {
350 PieCrd[i] = ps;
351 ps[0] = (float)(PieDiv - i - 1) / PieDiv;
352 ps[1] = (float)sqrt(1.0 - ps[0] * ps[0]);
353 }
354 float vp[4];
355 olx_gl::get(GL_VIEWPORT, vp);
356 TGXApp& app = TGXApp::GetInstance();
357 const TEBasis& basis = app.GetRenderer().GetBasis();
358 LinearScale = 1; // reset now
359 DrawScale = app.GetRenderer().GetBasis().GetZoom() / (LinearScale * app.GetRenderer().GetScale());
360 AradScale = 0.5 * DrawScale;
361 BondRad = 0.1 * DrawScale;
362 SceneOrigin = basis.GetCenter();
363 //DrawOrigin = vec3f(pw.GetWidth()/2, pw.GetHeight()/2, 0);
364 DrawOrigin = vec3f(vp[2] / 2, vp[3] / 2, 0);
365 ProjMatr = basis.GetMatrix() * DrawScale;
366 UnProjMatr = ProjMatr.Inverse();
367 }
368 void Render(const olxstr& fileName) {
369 TEXWriter pw(fileName);
370 Init(pw);
371 char bf[200];
372 bool CurrentAtom;
373
374 //temporary colour dictionary
375 //map<olxstr,string> ColourDic;
376 olxdict <olxstr, olxcstr, olxstrComparator<false> > ColourDic;
377 ColourDic("C", "gray");
378 ColourDic("N", "blue");
379 ColourDic("O", "red");
380 ColourDic("F", "darkgreen");
381 ColourDic("Si", "orange");
382 ColourDic("P", "purple");
383 ColourDic("S", "yellow");
384 olxcstr def_color = "white";
385
386 const TEBasis& basis = app.GetRenderer().GetBasis();
387 TTypeList<OrtDrawTex::OrtAtom> atoms;
388 TGXApp::AtomIterator ai = app.GetAtoms();
389 atoms.SetCapacity(ai.count);
390
391 //radius for the end of the bond
392 sprintf(bf, "\\newcommand{\\cornerradius}{%fmm}",
393 0.12 * 0.1 * DrawScale);
394 pw.Writenl(bf);
395
396 while (ai.HasNext()) {
397 TXAtom& sa = ai.Next();
398 if (sa.IsDeleted() || !sa.IsVisible())
399 continue;
400
401 //write latex colors atoms commands and shading
402 CurrentAtom = false;
403 //We check first if a similar atom already get processed.
404 for (size_t j = 0; j < atoms.Count(); j++) {
405 const TSAtom& sa2 = *atoms[j].atom;
406 if (sa.GetType() == sa2.GetType())
407 CurrentAtom = true;
408 }
409 //writing of the color definition of the atom
410 if (CurrentAtom == false)
411 {
412 sprintf(bf, "\\newcommand{\\color%s}{%s!90!black}",
413 sa.GetType().symbol.c_str(),
414 ColourDic.Find(sa.GetType().symbol, def_color).c_str()
415 );
416 pw.Writenl(bf);
417
418 sprintf(bf, "\\pgfdeclareradialshading{ballshading%s}{\\pgfpoint{-10bp}{10bp}}",
419 sa.GetType().symbol.c_str());
420 pw.Writenl(bf);
421 sprintf(bf, "{color(0bp)=(\\color%s!15!white); color(9bp)=(\\color%s!75!white);",
422 sa.GetType().symbol.c_str(), sa.GetType().symbol.c_str());
423 pw.Writenl(bf);
424 sprintf(bf, "color(18bp)=(\\color%s!70!black); color(25bp)=(\\color%s!50!black); color(50bp)=(black)} ",
425 sa.GetType().symbol.c_str(), sa.GetType().symbol.c_str());
426 pw.Writenl(bf);
427 }
428
429 if (sa.GetEllipsoid() != 0) {
430 mat3f& elpm = *(new mat3f(sa.GetEllipsoid()->GetMatrix()));
431 elpm.Scale(sa.GetEllipsoid()->GetNorms());
432 elpm *= TXAtom::GetSettings(app.GetRenderer()).GetTelpProb();
433 elpm *= ProjMatr;
434 mat3f& ielpm = *(new mat3f((sa.GetEllipsoid()->GetMatrix() * basis.GetMatrix()).Inverse()));
435 atoms.AddNew(&sa, (sa.crd() + SceneOrigin) * ProjMatr + DrawOrigin, &elpm, &(ielpm *= elpm));
436 }
437 else
438 atoms.AddNew(&sa, (sa.crd() + SceneOrigin) * ProjMatr + DrawOrigin);
439 }
440
441 //tex header is finished, start the body
442 pw.Writenl("\\begin{document}");
443 pw.Writenl("\\begin{tikzpicture}");
444 QuickSorter::SortSF(atoms, OrtAtomZSort);
445
446 for (size_t i = 0; i < atoms.Count(); i++) {
447 const TSAtom& sa = *atoms[i].atom;
448 const vec3d& p = atoms[i].crd;
449 if (sa.GetEllipsoid() == 0) {
450 //draw the atom as a circle, done twice, to make a white border arounf it
451 //draw circle, centre, radius
452 sprintf(bf, "\\begin{pgfscope}\\pgfsetstrokecolor{white}\\pgfsetlinewidth{\\whitespace}\\pgfpathcircle{\\pgfpoint{%fmm}{%fmm}}{%fmm}\n\\pgfshadepath{ballshading%s}{0}\n\\pgfusepath{draw}\\end{pgfscope}",
453 p[0] / 5, p[1] / 5,
454 AradScale * sqrt(sa.GetType().r_pers) / 5,
455 sa.GetType().symbol.c_str()
456 );
457 pw.Writenl(bf);
458 sprintf(bf, "\\pgfpathcircle{\\pgfpoint{%fmm}{%fmm}}{%fmm}\n\\pgfshadepath{ballshading%s}{0}\n\\pgfusepath{draw}",
459 p[0] / 5, p[1] / 5,
460 AradScale * sqrt(sa.GetType().r_bonding) / 5,
461 sa.GetType().symbol.c_str()
462 );
