45 template <short field> class TFileListItemSorter {
46 static olxstr ExtractFileExt(const olxstr& fn) {
47 size_t ind = fn.LastIndexOf('.');
48 return (ind != InvalidIndex) ? fn.SubStringFrom(ind+1) : EmptyString();
49 }
50 public:
51 template <class item_a_t, class item_b_t>
52 static int Compare(const item_a_t &i1_, const item_b_t &i2_) {
53 const TFileListItem &i1 = olx_ref::get(i1_),
54 &i2 = olx_ref::get(i2_);
55 if( field == 0 )
56 return i1.GetName().Compare(i2.GetName());
57 if( field == 1 ) {
58 return ExtractFileExt(i1.GetName()).Compare(
59 ExtractFileExt(i2.GetName()));
60 }
61 if( field == 2 )
62 return olx_cmp(i1.GetCreationTime(), i2.GetCreationTime());
63 if( field == 3 )
64 return olx_cmp(i1.GetModificationTime(), i2.GetModificationTime());
65 if( field == 4 )
66 return olx_cmp(i1.GetLastAccessTime(), i2.GetLastAccessTime());
67 if( field == 5 )
68 return olx_cmp(i1.GetSize(), i2.GetSize());
69 return 0;
70 }
71 };
