93struct cm_Gaussians {
94 double a1, a2, a3, a4, b1, b2, b3, b4, c;
95 cm_Gaussians() : a1(0), a2(0), a3(0), a4(0), b1(0), b2(0), b3(0), b4(0), c(0)
96 {}
97 // constructor, note that b values are inverted!
98 cm_Gaussians(double _a1, double _a2, double _a3, double _a4,
99 double _b1, double _b2, double _b3, double _b4, double _c) :
100 a1(_a1), a2(_a2), a3(_a3), a4(_a4),
101 b1(-_b1), b2(-_b2), b3(-_b3), b4(-_b4), c(_c)
102 {}
103 // copy constructor
104 cm_Gaussians(const cm_Gaussians& g) :
105 a1(g.a1), a2(g.a2), a3(g.a3), a4(g.a4),
106 b1(g.b1), b2(g.b2), b3(g.b3), b4(g.b4), c(g.c)
107 {}
108
109 double calc_sq(double sqv) const {
110 return a1*exp(b1*sqv) + a2*exp(b2*sqv) + a3*exp(b3*sqv) + a4*exp(b4*sqv) + c;
111 }
112 cm_Gaussians& operator = (const cm_Gaussians& g) {
113 a1 = g.a1; a2 = g.a2; a3 = g.a3; a4 = g.a4;
114 b1 = g.b1; b2 = g.b2; b3 = g.b3; b4 = g.b4;
115 c = g.c;
116 return *this;
117 }
118 bool operator == (const cm_Gaussians& g) const {
119 return (a1 == g.a1 && a2 == g.a2 && a3 == g.a3 && a4 == g.a4 &&
120 b1 == g.b1 && b2 == g.b2 && b3 == g.b3 && b4 == g.b4 &&
121 c == g.c);
122 }
123 bool operator != (const cm_Gaussians& g) const { return !(*this == g); }
124};

127protected:
128 double Mr;
129 short index;
130public:
131 const olxstr symbol, name;
132 uint32_t def_color;
133 const short z, isotope_count, henke_count;
134 double r_pers, r_cov, r_vdw, r_bonding, r_sfil;
135 // custom radii are to be used by procedure which require custom radii...
136 mutable double r_custom;
137 const cm_Gaussians* gaussians; // 9 elements = 4 gaussians + const
138 const cm_Isotope* isotopes;
139 const cm_Anomalous_Henke* henke_data;
140 const cm_Neutron_Scattering* neutron_scattering;
141 cm_Element(const char* _symbol, const char* _name,
142 uint32_t _def_color, short _z, double _r_vdw, double _r_cov, double _r_pers,
143 double _r_bonding, double _r_sfil, const cm_Gaussians* _gaussians,
144 const cm_Isotope* _isotopes, short _isotope_count,
145 const cm_Anomalous_Henke* _henke_data, short _henke_count,
146 const cm_Neutron_Scattering* _neutron_scattering) :
147 Mr(0), index(-1),
148 symbol(_symbol), name(_name), def_color(_def_color), z(_z),
149 isotope_count(_isotope_count), henke_count(_henke_count), r_pers(_r_pers),
150 r_cov(_r_cov), r_vdw(_r_vdw), r_bonding(_r_bonding), r_sfil(_r_sfil),
151 r_custom(0), gaussians(_gaussians), isotopes(_isotopes),
152 henke_data(_henke_data), neutron_scattering(_neutron_scattering)
153 {
154 if (z == 1 && symbol.CharAt(0) == 'D') {
155 Mr = isotopes[1].Mr;
156 }
157 else {
158 for (int i = 0; i < isotope_count; i++) {
159 Mr += isotopes[i].Mr * isotopes[i].W;
160 }
161 }
162 }
163
164 compd CalcFpFdp(double eV) const {
165 if (henke_data == NULL) {
166 throw TFunctionFailedException(__OlxSourceInfo,
167 "undefined f\' f\" data");
168 }
169 if (eV < henke_data[0].energy || eV > henke_data[henke_count - 1].energy) {
170 throw TInvalidArgumentException(__OlxSourceInfo,
171 "energy is out of range");
172 }
173 for (int i = 0; i < henke_count; i++) {
174 if (henke_data[i].energy == eV)
175 return compd(henke_data[i].fp, henke_data[i].fdp);
176 if (henke_data[i].energy > eV) {
177 double k = (eV - henke_data[i - 1].energy) /
178 (henke_data[i].energy - henke_data[i - 1].energy);
179 double fp = cm_Anomalous_Henke::Undefined;
180 double fdp = cm_Anomalous_Henke::Undefined;
181 if (henke_data[i - 1].fp != cm_Anomalous_Henke::Undefined &&
182 henke_data[i].fp != cm_Anomalous_Henke::Undefined)
183 {
184 fp = henke_data[i - 1].fp + k*(henke_data[i].fp - henke_data[i - 1].fp);
185 }
186 if (henke_data[i - 1].fdp != cm_Anomalous_Henke::Undefined &&
187 henke_data[i].fdp != cm_Anomalous_Henke::Undefined)
188 {
189
190 fdp = henke_data[i - 1].fdp + k*(henke_data[i].fdp - henke_data[i - 1].fdp);
191 }
192 return compd(fp, fdp);
193 }
194 }
195 return compd();
196 }
197 double GetMr() const { return Mr; }

262 }

269 return ToString(*element, count, charge);

272 static olxstr ToString(const cm_Element &element, double count, int charge) {
273 olxstr rv;
274 if (charge != 0) {
275 rv << '(' << element.symbol << olx_sign_char(charge);
276 int ch = olx_abs(charge);
277 if (ch > 1) {
278 rv << ch;
279 }
280 rv << ')';
281 }
282 else {
283 rv = element.symbol;
284 }
285 return rv << olxstr::FormatFloat(3, count).TrimFloat();
286 }

389 /* checks if given element is in sub-group 1-10, Lanthanised are

391 static bool IsTtransitionalGroup(int group, const cm_Element &e) {
392 if( e.z >= 21 && e.z <= 30 ) { // Sc->Zn
393 int z = e.z > 28 ? e.z-10 : e.z;
394 return z == (group+18);
395 }
396 else if( e.z >= 39 && e.z <= 48 ) { //Y->Cd
397 int z = e.z > 46 ? e.z-10 : e.z;
398 return z == (group+36);
399 }
400 else if( e.z >= 57 && e.z <= 80 ) { //La->Hg
401 if( e.z < 72 ) // Lanthanides
402 return group == 3;
403 int z = e.z > 78 ? e.z-10 : e.z;
404 return z == (group+68);
405 }
406 return false;
407 }
