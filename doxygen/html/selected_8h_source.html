45class AtomTypeMask : public IAtomMask {
46 struct AtomType {
47 short z;
48 const cm_Element *element;
49 bool Not;
50 AtomType(short z, bool Not) : z(z), element(0), Not(Not)
51 {}
52 AtomType(const cm_Element &elm, bool Not) : z(0), element(&elm), Not(Not)
53 {}
54 int Compare(const AtomType &a) const {
55 if (element == 0) {
56 if (a.element == 0)
57 return olx_cmp(z, a.z);
58 if (z == -2 && XElementLib::IsMetal(*a.element))
59 return 0;
60 if (z == -3 && XElementLib::IsHalogen(*a.element))
61 return 0;
62 return -1;
63 }
64 else {
65 if (a.element == 0)
66 return a.Compare(*this);
67 return olx_cmp(element->z, a.element->z);
68 }
69 }
70 olxstr str() const {
71 if (element != 0) return element->symbol;
72 if (z == -2) return 'M';
73 return 'X';
74 }
75 olxstr toString() const {
76 return Not ? str().Insert('-', 0) : str();
77 }
78 };

80 bool Not;
81public:
