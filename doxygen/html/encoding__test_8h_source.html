114 void Base85Test(OlxTests& t) {
115 t.description = __FUNC__;
116 olxstr msgs[] = { L"The quick brown fox",
117 L"%% %%% % \u2045" };
118 uint8_t test_data_1[8] = {0x86, 0x4F, 0xD2, 0x6F, 0xB5, 0x59, 0xF7, 0x5B};
119 olxcstr x1 = encoding::base85::decode("HelloWorld");
120 if (x1.Length() != 8) {
121 throw TFunctionFailedException(__OlxSourceInfo, "encoding failed");
122 }
123 for (int i = 0; i < 8; i++) {
124 if (test_data_1[i] != (uint8_t)x1.CharAt(i)) {
125 throw TFunctionFailedException(__OlxSourceInfo, "encoding failed");
126 }
127 }
128 olxcstr x = encoding::base85::encode(&test_data_1[0], 8);
129 if (x != "HelloWorld") {
130 throw TFunctionFailedException(__OlxSourceInfo, "encoding failed");
131 }
132 for (int i = 0; i < 2; i++) {
133 olxcstr toe = TUtf8::Encode(msgs[i]);
134 olxcstr eres = encoding::base85::encode(toe);
135 olxcstr tod = encoding::base85::decode(eres);
136 olxstr dres = TUtf8::Decode(tod);
137 if (dres != msgs[i]) {
138 throw TFunctionFailedException(__OlxSourceInfo,
139 "Non reversble encoding");

144 void MBStest(OlxTests &t) {
145 t.description = __FUNC__;
146 olxstr cm[] = { L"The quick brown fox",
147 L"\u00e9 \u00df \u00de "
148 },
149 fm[] = {L"%% \u2045", L"\u06de \u20a9"};
150 uint32_t fu[2][3] = { {25,0x24B62, 0x10437 }, {0x06de, 32, 0x20a9} };
151 for (int i = 0; i < 2; i++) {
152 olxcstr e = cm[i].ToMBStr();
153 olxstr d = e.ToWCStr();
154 if (d != cm[i]) {
155 throw TFunctionFailedException(__OlxSourceInfo, "MBS-WStr conversion failed");
156 }
157 }
158 for (int i = 0; i < 2; i++) {
159 try {
160 olxcstr e = fm[i].ToMBStr();
161 olxstr d = e.ToWCStr();
162 throw TFunctionFailedException(__OlxSourceInfo, "MBS-WStr did not fail but should have");
163 }
164 catch (const TExceptionBase &e) {
165 }
166 }
167 for (int i = 0; i < 2; i++) {
168 olxcstr e = fm[i].ToUTF8(), e1;
169 e1 = e;
170 if (!e.IsUTF8() || !e.SubString(0, 1).IsUTF8() || !olxcstr(e).IsUTF8() || !e1.IsUTF8()) {
171 throw TFunctionFailedException(__OlxSourceInfo, "UTF8 flag is not set or not copied over");
172 }
173 olxstr d = e.ToWCStr();
174 if (d != fm[i] || d != TUtf8::Decode(e)) {
175 throw TFunctionFailedException(__OlxSourceInfo, "UTF8 conversion failed");
176 }
177 }
178 for (int i = 0; i < 2; i++) {
179 TArrayList<uint16_t> x = TUtf16::Encode(&fu[i][0], 3);
180 if (TUtf16::CharCount(x.GetData(), x.Count()) != 3) {
181 throw TFunctionFailedException(__OlxSourceInfo, "UTF16 conversion failed");
182 }
183 TArrayList<uint32_t> y = TUtf16::Decode(x.GetData(), x.Count());
184 if (y.Count() != 3) {
185 throw TFunctionFailedException(__OlxSourceInfo, "UTF16 conversion failed");
186 }
187 for (int j = 0; j < 3; j++) {
188 if (fu[i][j] != y[j]) {
189 throw TFunctionFailedException(__OlxSourceInfo, "UTF16 conversion failed");
190 }
191 }
192 }
193 }
