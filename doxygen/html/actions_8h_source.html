41class AActionHandler : public ACollectionItem {
42 bool Enabled, // not executed if false
43 ToDelete; // default is true - the object will be deleted by the collection!!
44public:
45 AActionHandler() : Enabled(true), ToDelete(true) {}
46 virtual ~AActionHandler() {}
47 // handler before the event: sender, data, parent queue
48 virtual bool Enter(const IOlxObject *, const IOlxObject *, TActionQueue *) {
49 return false;
50 }
51 // handler after the event: sender, data, parent queue
52 virtual bool Exit(const IOlxObject *, const IOlxObject *, TActionQueue *) {
53 return false;
54 }
55 /* handler during the event, does some operation when Obj is changed;
56 the action depends on the type of queue: sender, data, parent queue
57 */
58 virtual void UpdateData(const IOlxObject *, const IOlxObject *, TActionQueue *)
59 {}
60 //: sender, data, parent queue
61 virtual bool Execute(const IOlxObject *, const IOlxObject *, TActionQueue *) {
62 return false;
63 }
64 // the function is called the object is removed from the queue
65 virtual void OnRemove(TActionQueue *) {}
66 DefPropBIsSet(Enabled)
67 /* beware that it is set to true by default, which means that the object will
68 be automatically deleted by the parent queue
69 */
70 DefPropBIsSet(ToDelete)
71};

217class TActionQueueLock {
218 TActionQueue* queue;
219 bool changed;
220public:
221 TActionQueueLock(TActionQueue* _queue) : queue(_queue), changed(false) {
222 if (queue != NULL) {
223 if (queue->IsEnabled()) {
224 changed = true;
225 queue->SetEnabled(false);
226 }
227 }
228 }
229 ~TActionQueueLock() {
230 if (changed && queue != NULL)
231 queue->SetEnabled(true);
232 }
233 void Unlock() {
234 if (changed) {
235 if (queue != NULL)
236 queue->SetEnabled(true);
237 changed = false;
238 }
239 }
240};
