79 template <class RefList, class MatList>
80 void ExpandToP1(const RefList& hkl_list, const TArrayList<compd>& F,
81 const MatList& ml,
82 TArrayList<StructureFactor>& out_)
83 {
84 const size_t ml_cnt = ml.Count();
85 TTypeList<StructureFactor> tmp(ml_cnt * hkl_list.Count());
86 for (size_t i = 0; i < hkl_list.Count(); i++) {
87 const size_t off = i*ml_cnt;
88 for (size_t j = 0; j < ml_cnt; j++) {
89 const smatd& m = ml[j];
90 const size_t ind = off + j;
91 vec3i hkl = TReflection::GetHkl(hkl_list[i]);
92 tmp[ind].hkl = hkl*m.r;
93 tmp[ind].ps = m.t.DotProd(hkl);
94 if (tmp[ind].ps != 0) {
95 double ca = 1, sa = 0;
96 olx_sincos(-T_PI*tmp[ind].ps, &sa, &ca);
97 tmp[ind].val = F[i] * compd(ca, sa);
98 }
99 else {
100 tmp[ind].val = F[i];
101 }
102 }
103 }
104 QuickSorter::Sort(tmp, TComparableComparator());
105 for (size_t i = 0; i < tmp.Count(); i++) {
106 size_t j = i;
107 while (++j < tmp.Count() && tmp[i].hkl == tmp[j].hkl) {
108 tmp.NullItem(j);
109 }
110 i = j - 1;
111 }
112 tmp.Pack();
113 out_ = tmp;
114 }

198 SFCalculateTask(const SF_Util& _parent, const IMillerIndexList& _refs,
199 const mat3d& _hkl2c,
200 TArrayList<compd>& _F,
201 const TTypeList<XScatterer>& _scatterers,
202 const TCAtomPList& _atoms, const double* _U)
203 : parent(_parent), refs(_refs), hkl2c(_hkl2c), F(_F),
204 scatterers(_scatterers), atoms(_atoms), U(_U),
205 rv(_parent._getusize()), ps(_parent._getusize()),
206 fo(_scatterers.Count())
207 {}
208 virtual ~SFCalculateTask() {}
209 double calc_B(const double* Q, const vec3i &h) const {
210 return exp(
211 (Q[0] * h[0] + Q[4] * h[2] + Q[5] * h[1])*h[0] +
212 (Q[1] * h[1] + Q[3] * h[2])*h[1] +
213 (Q[2] * h[2])*h[2]);
214 }
215 void Run(size_t i) {
216 const vec3i hkl = refs[i]; //make a copy, safer
217 const double d_s2 = TReflection::ToCart(hkl, hkl2c).QLength()*0.25;
218 parent._generateu(hkl, rv, ps);
219 if (!anom_only) {
220 for (size_t j = 0; j < scatterers.Count(); j++) {
221 fo[j] = scatterers[j].calc_sq(d_s2);
222 }
223 }
224 if (centro) {
225 compd ir = 0;
226 for (size_t j = 0; j < atoms.Count(); j++) {
227 double l = 0;
228 for (size_t k = 0; k < parent._getusize(); k++) {
229 // scattering vector + phase shift
230 const double tv = SFUtil::T_PI*(atoms[j]->ccrd().DotProd(rv[k]) + ps[k]);
231 double ca, sa;
232 olx_sincos(tv, &sa, &ca);
233 if (olx_is_valid_index(atoms[j]->GetEllpId())) {
234 const double B = calc_B(&U[j * 6], rv[k]);
235 if (atoms[j]->GetEllipsoid()->IsAnharmonic()) {
236 l += (atoms[j]->GetEllipsoid()->
237 GetAnharmonicPart()->Calculate(rv[k])*compd(ca*B, sa*B)).GetRe();
238 }
239 else {
240 l += ca * B;
241 }
242 }
243 else {
244 l += ca;
245 }
246 }
247 const Disp* disp = &atoms[j]->GetDisp();
248 compd scv = fo[atoms[j]->GetTag()] +
249 (disp == 0 ? scatterers[atoms[j]->GetTag()].GetFpFdp() : disp->value);
250 if (!olx_is_valid_index(atoms[j]->GetEllpId())) {
251 scv *= exp(U[j * 6] * d_s2);
252 }
253 scv *= atoms[j]->GetOccu();
254 scv *= l;
255 ir += scv;
256 }
257 F[i] = ir*parent._getumult();
258 }
259 else {
260 compd ir = 0;
261 for (size_t j = 0; j < atoms.Count(); j++) {
262 compd l = 0;
263 for (size_t k = 0; k < parent._getusize(); k++) {
264 // scattering vector + phase shift
265 const double tv = SFUtil::T_PI*(atoms[j]->ccrd().DotProd(rv[k]) + ps[k]);
266 double ca, sa;
267 olx_sincos(tv, &sa, &ca);
268 if (olx_is_valid_index(atoms[j]->GetEllpId())) {
269 const double B = calc_B(&U[j * 6], rv[k]);
270 if (atoms[j]->GetEllipsoid()->IsAnharmonic()) {
271 l += atoms[j]->GetEllipsoid()->
272 GetAnharmonicPart()->Calculate(rv[k])*compd(ca*B, sa*B);
273 }
274 else {
275 l.Re() += ca * B;
276 l.Im() += sa * B;
277 }
278 }
279 else {
280 l.Re() += ca;
281 l.Im() += sa;
282 }
283 }
284 const Disp* disp = &atoms[j]->GetDisp();
285 compd scv = fo[atoms[j]->GetTag()] +
286 (disp == 0 ? scatterers[atoms[j]->GetTag()].GetFpFdp() : disp->value);
287 if (!olx_is_valid_index(atoms[j]->GetEllpId())) {
288 scv *= exp(U[j * 6] * d_s2);
289 }
290 scv *= atoms[j]->GetOccu();
291 scv *= l;
292 ir += scv;
293 }
294 F[i] = ir*parent._getumult();
295 }
296 }

306 virtual void Expand(const TArrayList<vec3i>& hkl, const TArrayList<compd>& F,
307 TArrayList<SFUtil::StructureFactor>& out) const
308 {
309 TArrayList<vec3i> rv(sg::size);
310 TArrayList<double> ps(sg::size);
311 const size_t hkl_cnt = hkl.Count();
312 for (size_t i = 0; i < hkl_cnt; i++) {
313 sg::GenHkl(hkl[i], rv, ps);
314 const size_t off = i*sg::size;
315 for (size_t j = 0; j < sg::size; j++) {
316 const size_t ind = j + off;
317 out[ind].hkl = rv[j];
318 out[ind].ps = ps[j];
319 double ca = 1, sa = 0;
320 if (ps[j] != 0) {
321 olx_sincos(-SFUtil::T_PI*ps[j], &sa, &ca);
322 out[ind].val = F[i] * compd(ca, sa);
323 }
324 else {
325 out[ind].val = F[i];
326 }
327 }
328 }
329 QuickSorter::Sort(out, TComparableComparator());
330 for (size_t i = 0; i < out.Count(); i++) {
331 size_t j = i;
332 while (++j < out.Count() && out[i].hkl == out[j].hkl) {
333 out.Delete(j--);
334 }
335 }
