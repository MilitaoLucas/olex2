32template <class Impl> struct ThreadSingletonMainInstance {
33 Impl Instance;
34 olx_pdict<unsigned long, Impl*> Instances;
35 unsigned long main_thread_id;
36 ThreadSingletonMainInstance()
37 : main_thread_id(AOlxThread::GetCurrentThreadId())
38 {}
39 ~ThreadSingletonMainInstance() {
40 for (int i=0; i < Instances.Count(); i++)
41 delete Instances.GetValue(i);
42 Instances.Clear();
43 }
44};
45template <class Impl> class ThreadSingletonBase {
46protected:
47 static ThreadSingletonMainInstance<Impl> &Instance() {
48 static static ThreadSingletonMainInstance<Impl> i;
49 return i;
50 }
51 static olx_critical_section cs;
52 ThreadSingletonBase() : Impl() {}
53 virtual ~ThreadSingletonBase() {}
54public:
55 static Impl& GetInstance() {
56 if (Instance().main_thread_id == AOlxThread::GetCurrentThreadId())
57 return Instance().GetInstance();
58 else {
59 unsigned long th_id = AOlxThread::GetCurrentThreadId();
60 cs.enter();
61 Impl* rv = NULL;
62 try {
63 size_t ii = Instance().Instances.IndexOf(th_id);
64 if (ii != InvalidIndex)
65 rv = &Instance().Instances.GetValue(ii)->GetInstance();
66 else
67 rv = &Instance().Instances.Add(th_id, new Impl)->GetInstance();
68 }
69 catch(...) {}
70 cs.leave();
71 if (rv == NULL) {
72 throw TFunctionFailedException(__OlxSourceInfo,
73 "could not acquire object instance");
74 }
75 return *rv;
76 }
77 }
78 static bool HasInstance() {
79 unsigned long th_id = AOlxThread::GetCurrentThreadId();
80 if( th_id == Instance().main_thread_id )
81 return true;
82 else
83 return Instance.Instances.HasKey(th_id);
84 }
85};
