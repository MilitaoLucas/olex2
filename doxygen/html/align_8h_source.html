56 /* finds allignment quaternions for given coordinates and their weights
57 the rmsds (and the quaternions) are sorted ascending
58 Acta A45 (1989), 208.
59 The resulting matrices map {b} set to {a} */
60 template <class List> out FindAlignmentQuaternions(const List& pairs,
61 bool update_centers=true)
62 {
63 out ao;
64 double swa = 0, swb = 0, sws = 0;
65 for (size_t i = 0; i < pairs.Count(); i++) {
66 if (update_centers) {
67 ao.center_a += pairs[i].GetValueA()*pairs[i].GetWeightA();
68 ao.center_b += pairs[i].GetValueB()*pairs[i].GetWeightB();
69 swa += pairs[i].GetWeightA();
70 swb += pairs[i].GetWeightB();
71 }
72 sws += pairs[i].GetWeightA()*pairs[i].GetWeightB();
73 }
74 if (update_centers) {
75 ao.center_a /= swa;
76 ao.center_b /= swb;
77 }
78 ematd evm(4, 4);
79 for (size_t i = 0; i < pairs.Count(); i++) {
80 const vec3d v1 = (pairs[i].GetValueA() - ao.center_a)*pairs[i].GetWeightA();
81 const vec3d v2 = (pairs[i].GetValueB() - ao.center_b)*pairs[i].GetWeightB();
82 const vec3d p = v1 + v2;
83 const vec3d m = v1 - v2;
84 evm[0][0] += (m[0] * m[0] + m[1] * m[1] + m[2] * m[2]);
85 evm[0][1] += (p[1] * m[2] - m[1] * p[2]);
86 evm[0][2] += (m[0] * p[2] - p[0] * m[2]);
87 evm[0][3] += (p[0] * m[1] - m[0] * p[1]);
88 evm[1][1] += (p[1] * p[1] + p[2] * p[2] + m[0] * m[0]);
89 evm[1][2] += (m[0] * m[1] - p[0] * p[1]);
90 evm[1][3] += (m[0] * m[2] - p[0] * p[2]);
91 evm[2][2] += (p[0] * p[0] + p[2] * p[2] + m[1] * m[1]);
92 evm[2][3] += (m[1] * m[2] - p[1] * p[2]);
93 evm[3][3] += (p[0] * p[0] + p[1] * p[1] + m[2] * m[2]);
94 }
95 evm[1][0] = evm[0][1];
96 evm[2][0] = evm[0][2];
97 evm[2][1] = evm[1][2];
98 evm[3][0] = evm[0][3];
99 evm[3][1] = evm[1][3];
100 evm[3][2] = evm[2][3];
101 ematd::EigenValues(evm /= sws, ao.quaternions.I());
102 for (int i = 0; i < 4; i++) {
103 ao.rmsd[i] = (evm[i][i] <= 0 ? 0 : sqrt(evm[i][i]));
104 }
105 bool changes = true;
106 while (changes) {
107 changes = false;
108 for (int i = 0; i < 3; i++) {
109 if (ao.rmsd[i + 1] < ao.rmsd[i]) {
110 ao.quaternions.SwapRows(i, i + 1);
111 olx_swap(ao.rmsd[i], ao.rmsd[i + 1]);
112 changes = true;
113 }
114 }
115 }
116 return ao;
117 }

119 template <class List> double CalcRMSD(const List& pairs,
120 const align::out& ao)
121 {
122 double rmsd = 0;
123 mat3d m;
124 QuaternionToMatrix(ao.quaternions[0], m);
125 for (size_t i = 0; i < pairs.Count(); i++) {
126 rmsd += (pairs[i].GetValueA() - ao.center_a)
127 .QDistanceTo((pairs[i].GetValueB() - ao.center_b)*m);
128 }
129 return sqrt(rmsd / pairs.Count());
130 }

211 struct out {
212 TVector3<FloatT> moments, center;
213 TMatrix33<FloatT> axes;
214 void sort() {
215 bool swaps = true;
216 while (swaps) {
217 swaps = false;
218 for (int i = 0; i < 2; i++) {
219 if (moments[i] > moments[i + 1]) {
220 olx_swap(axes[i], axes[i + 1]);
221 olx_swap(moments[i], moments[i + 1]);
222 swaps = true;
223 }
224 }
225 }
226 // keep the orientation system
227 axes[2] = axes[0].XProdVec(axes[1]).Normalise();
228 }

231 template <class list_t, class p_accessor_t, class w_accessor_t>
232 static out calc(const list_t &list,
233 const p_accessor_t &p_accessor,
234 const w_accessor_t &w_accessor)
235 {
236 out rv;
237 if (list.Count() < 2) {
238 return rv;
239 }
240 FloatT w_sum = 0;
241 for (size_t i = 0; i < list.Count(); i++) {
242 FloatT w = w_accessor(list[i]);
243 rv.center += p_accessor(list[i])*w;
244 w_sum += w;
245 }
246 rv.center /= w_sum;
247 TMatrix33<FloatT> I;
248 for (size_t i = 0; i < list.Count(); i++) {
249 TVector3<FloatT> v =
250 (p_accessor(list[i]) - rv.center)*w_accessor(list[i]);
251 I[0][0] += (olx_sqr(v[1]) + olx_sqr(v[2]));
252 I[1][1] += (olx_sqr(v[0]) + olx_sqr(v[2]));
253 I[2][2] += (olx_sqr(v[0]) + olx_sqr(v[1]));
254 I[0][1] -= v[0] * v[1];
255 I[0][2] -= v[0] * v[2];
256 I[1][2] -= v[1] * v[2];
257 }
258 I[1][0] = I[0][1];
259 I[2][0] = I[0][2];
260 I[2][1] = I[1][2];
261 TMatrix33<FloatT>::EigenValues(I, rv.axes.I());
262 for (int i = 0; i < 3; i++) {
263 rv.moments[i] = I[i][i] < 0 ? 0 : sqrt(I[i][i]);
264 }
265 return rv;
266 }

268 template <class list_t, class accessor_t>
269 static out calc(const list_t &list, const accessor_t &accessor) {
270 return calc(list, accessor, unit_weight());
271 }
