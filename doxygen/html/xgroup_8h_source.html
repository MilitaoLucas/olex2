45 virtual void DoDraw(bool SelectPrimitives, bool SelectObjects) {
46 if (GetParentGroup() != 0) { // is inside a group?
47 TGlGroup::DoDraw(SelectPrimitives, SelectObjects);
48 return;
49 }
50 for (int ditr=0; ditr < 2; ditr++) {
51 for (size_t i = 0; i < Atoms.Count(); i++) {
52 olx_swap(Atoms[i]->crd(), src_crds[i]);
53 }
54 for (size_t i = 0; i < Bonds.Count(); i++) {
55 Bonds[i]->Update();
56 }
57 for (size_t i=0; i < TGlGroup::Count(); i++) {
58 AGDrawObject& G = GetObject(i);
59 if (!G.IsVisible()) {
60 continue;
61 }
62 if (G.IsGroup()) {
63 TGlGroup* group = dynamic_cast<TGlGroup*>(&G);
64 if (group != 0) {
65 group->Draw(SelectPrimitives, SelectObjects);
66 continue;
67 }
68 }
69 bool Select = (SelectObjects|SelectPrimitives);
70 const size_t pc = G.GetPrimitives().PrimitiveCount();
71 for (size_t j=0; j < pc; j++) {
72 TGlPrimitive& GlP = G.GetPrimitives().GetPrimitive(j);
73 if (!Select) {
74 bool initialised=false;
75 if (&G == rotation_anchor) {
76 if (G.Is<TXAtom>()) {
77 TXAtom &a = (TXAtom &)G;
78 if (a.crd().Equals(RotationCenter, 1e-3))
79 initialised = true;
80 }
81 else if (G.Is<TXBond>()) {
82 TXBond &b = (TXBond &)G;
83 if (RotationCenter.Equals((b.A().crd() + b.B().crd()) / 2, 1e-3)) {
84 initialised = true;
85 }
86 }
87 }
88 if (initialised) {
89 TGlGroup::GetGlM().Init(false);
90 }
91 else {
92 TGlMaterial glm = GlP.GetProperties();
93 glm.SetFlags(glm.GetFlags()|sglmShininessF|sglmSpecularF);
94 glm.AmbientF *= 0.75;
95 if (ditr == 1) {
96 glm.AmbientF = glm.AmbientF.GetRGB() | 0x007070;
97 }
98 else {
99 glm.AmbientF = glm.AmbientF.GetRGB() | 0x700070;
100 }
101 glm.ShininessF = 32;
102 glm.SpecularF = 0xff00;
103 glm.Init(false);
104 }
105 }
106 Parent.HandleSelection(G, GlP, SelectObjects, SelectPrimitives);
107 olx_gl::pushMatrix();
108 if (G.Orient(GlP)) {
109 olx_gl::popMatrix();
110 continue;
111 }
112 GlP.Draw();
113 olx_gl::popMatrix();
114 }
115 }
116 }

119 virtual bool DoTranslate(const vec3d& t) {
120 for (size_t i = 0; i < Atoms.Count(); i++) {
121 Atoms[i]->crd() += t;
122 if (mirror) {
123 src_crds[i] -= t;
124 }
125 }
126 RotationCenter += t;
127 for (size_t i = 0; i < Count(); i++) {
128 GetObject(i).Update();
129 }
130 return true;
131 }
132
133 virtual bool DoRotate(const vec3d& vec, double angle) {
134 if (AngleInc != 0) {
135 if (olx_abs(AngleAcc += angle) < AngleInc) {
136 return true;
137 }
138 angle = AngleInc*olx_sign(AngleAcc);
139 AngleAcc = 0;
140 }
141 mat3d m;
142 if (RotationDir.IsNull()) {
143 olx_create_rotation_matrix(m, vec, cos(angle), sin(angle));
144 }
145 else {
146 olx_create_rotation_matrix(m, RotationDir, cos(angle), sin(angle));
147 }
148 for (size_t i = 0; i < Atoms.Count(); i++) {
149 Atoms[i]->crd() = (Atoms[i]->crd() - RotationCenter)*m + RotationCenter;
150 if (mirror) {
151 src_crds[i] = m*(src_crds[i] - RotationCenter) + RotationCenter;
152 }
153 }
154 for (size_t i = 0; i < Bonds.Count(); i++) {
155 Bonds[i]->Update();
156 }
157 return true;
158 }

170 bool OnMouseUp(const IOlxObject *Sender, const TMouseData& Data) {
171 if (Data.Button == smbRight) {
172 if (Data.HasObject()) {
173 if (Data.GetObject()->Is<TXAtom>()) {
174 if (rotation_anchor == Data.GetObject()) {
175 rotation_anchor = 0;
176 UpdateRotationCenter();
177 }
178 else {
179 rotation_anchor = Data.GetObject();
180 RotationCenter = ((TXAtom*)Data.GetObject())->crd();
181 RotationDir.Null();
182 }
183 }
184 else if (Data.GetObject()->Is<TXBond>()) {
185 if (rotation_anchor == Data.GetObject()) {
186 rotation_anchor = 0;
187 RotationDir.Null();
188 UpdateRotationCenter();
189 }
190 else {
191 rotation_anchor = Data.GetObject();
192 TXBond* xb = (TXBond*)rotation_anchor;
193 RotationCenter = (xb->A().crd() + xb->B().crd()) / 2;
194 RotationDir = (xb->B().crd() - xb->A().crd()).Normalise();
195 }
196 }
197 return true;
198 }
199 }
200 return GetHandler().OnMouseUp(*this, Data);
201 }

229 void AddAtoms(const TPtrList<TXAtom>& atoms) {
230 TGlGroup::AddObjects(atoms);
231 Atoms.SetCapacity(atoms.Count());
232 src_crds.SetCount(atoms.Count());
233 for (size_t i = 0; i < atoms.Count(); i++) {
234 Bonds.AddAll(Atoms.Add(atoms[i])->GetBonds(),
235 StaticCastAccessor<TXBond>());
236 src_crds[i] = atoms[i]->crd();
237 }
238 Bonds.ForEach(ACollectionItem::IndexTagSetter());
239 Bonds.Pack(
240 olx_alg::olx_or(
241 olx_alg::olx_not(ACollectionItem::IndexTagAnalyser()),
242 AGDrawObject::FlagsAnalyser(sgdoHidden)
243 )
244 );
245 TGlGroup::AddObjects(Bonds);
246 UpdateRotationCenter();
247 }

256 void Update() {
257 Atoms.Clear();
258 Bonds.Clear();
259 src_crds.Clear();
260 src_crds.SetCapacity(TGlGroup::Count());
261 for (size_t i = 0; i < TGlGroup::Count(); i++) {
262 AGDrawObject& G = GetObject(i);
263 if (G.Is<TXAtom>()) {
264 src_crds.Add(Atoms.Add((TXAtom&)G)->crd());
265 }
266 else if (G.Is<TXBond>()) {
267 Bonds.Add((TXBond&)G);
268 }
269 }

272 void UpdateRotationCenter() {
273 RotationCenter.Null();
274 for (size_t i = 0; i < Atoms.Count(); i++) {
275 RotationCenter += Atoms[i]->crd();
276 }
277 RotationCenter /= Atoms.Count();
278 }
