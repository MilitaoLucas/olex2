56 class handler {
57 public:
58 struct iterator {
59 const handler& obs;
60 size_t h_idx;
61 mutable size_t current;
62 mutable double scale;
63 iterator(const handler& parent, size_t h_idx)
64 : obs(parent), h_idx(h_idx),
65 current(InvalidIndex), scale(0)
66 {}
67 bool has_next() const {
68 return current == InvalidIndex || obs.components[h_idx].Count() > current;
69 }
70 const vec3i& next_index() const {
71 if (++current == 0) {
72 scale = obs.scales[olx_abs(obs.measured[h_idx].GetBatch()) - 1].value;
73 return obs.measured[h_idx].GetHkl();
74 }
75 else {
76 scale = obs.components[h_idx][current - 1].scale.value;
77 return obs.components[h_idx][current - 1].index;
78 }
79 }
80 const TReflection& next_obs() const {
81 vec3i hkl = next_index();
82 return obs.measured[obs.find_obs(hkl)];
83 }
84 };
85 struct twin_mate_generator {
86 const iterator& itr;
87 const TArrayList<compd>& Fc;
88 twin_mate_generator(const iterator& _itr, const TArrayList<compd>& _Fc)
89 : itr(_itr), Fc(_Fc)
90 {}
91 bool has_next() const { return itr.has_next(); }
92 olx_object_ptr<twin_mate_full> next_full() const {
93 vec3i hkl = itr.next_index();
94 const map_et& r = itr.obs.hkl_to_ref_map->Value(hkl);
95 return new twin_mate_full(
96 Fc[r.a],
97 itr.obs.measured[itr.h_idx].GetI(),
98 itr.obs.measured[itr.h_idx].GetS(),
99 itr.scale);
100 }
101 twin_mate_calc next_calc() const {
102 vec3i hkl = itr.next_index();
103 return twin_mate_calc(Fc[itr.obs.find_calc(hkl)], itr.scale);
104 }
105 };

127 template <typename detwinner_t>
128 void detwin(const detwinner_t& dt, TRefList& out,
129 const TArrayList<compd>& Fc)
130 {
131 out = measured;
132 for (size_t i = 0; i < out.Count(); i++) {
133 TReflection& r = out[i];
134 detwin_result res = dt.detwin(twin_mate_generator(iterate(i), Fc));
135 r.SetI(res.f_sq);
136 r.SetS(res.sig);
137 r.SetBatch(TReflection::NoBatchSet);
138 }
139 }

141 template <typename detwinner_t, typename merger_t>
142 void detwin_and_merge(const detwinner_t& dt, const merger_t& merger,
143 TRefList& out, const TArrayList<compd>& Fc, TArrayList<compd>* pF)
144 {
145 detwin(dt, out, Fc);
146 TRefPList to_merge(out);
147 out.ReleaseAll();
148 SymmSpace::InfoEx si = sym_info;
149 si.centrosymmetric = true;
150 RefMerger::Merge<merger_t>(sym_info, to_merge, out, vec3i_list());
151 to_merge.DeleteItems(false);
152 if (pF != 0) {
153 TArrayList<compd>& F = *pF;
154 olx_array::TArray3D<map_et>& map = *hkl_to_ref_map;
155 F.SetCount(out.Count());
156 for (size_t i = 0; i < out.Count(); i++) {
157 size_t f_i;
158 if (!map.IsInRange(out[i].GetHkl()) ||
159 (f_i = map(out[i].GetHkl()).a) == InvalidIndex)
160 {
161 throw TFunctionFailedException(__OlxSourceInfo,
162 "merging does not match");
163 }
164 F[i] = Fc[f_i];
165 }
166 }
167 }
168 void calc_fsq(const TArrayList<compd>& Fc, evecd& Fsq) {
169 Fsq.Resize(measured.Count());
170 for (size_t i = 0; i < measured.Count(); i++) {
171 Fsq[i] = calc_f_sq_1(twin_mate_generator(iterate(i), Fc));
172 }
173 }
174
175 protected:
176 double calc_f_sq_1(const twin_mate_generator& itr) {
177 double res = itr.next_calc().f_sq_calc();
178 while (itr.has_next()) {
179 res += itr.next_calc().f_sq_calc();
180 }
181 return res;
182 }
