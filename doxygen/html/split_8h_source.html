16enum {

20class TSplitMode : public AEventsDispatcher, public AMode {
21 TTypeList<olx_pair_t<TCAtom*, TCAtom*> > SplitAtoms;
22protected:
23 void UpdateSelectionCrds() const {
24 TGlGroup& sel = gxapp.GetSelection();
25 if (sel.Count() > 1) {
26 vec3d cr;
27 TXAtomPList atoms;
28 for (size_t i = 0; i < sel.Count(); i++) {
29 if (sel[i].Is<TXAtom>()) {
30 cr += ((TXAtom&)sel[i]).GetCenter();
31 cr += ((TXAtom&)sel[i]).crd();
32 atoms.Add((TXAtom&)sel[i]);
33 }
34 }
35 }
36 }
37 void UpdateCrds() const {
38 const TAsymmUnit& au = gxapp.XFile().GetAsymmUnit();
39 UpdateSelectionCrds();
40 TGXApp::AtomIterator ai = gxapp.GetAtoms();
41 while (ai.HasNext()) {
42 TXAtom& xa = ai.Next();
43 // summ the translations
44 xa.crd() += xa.GetCenter();
45 xa.NullCenter();
46 vec3d c = xa.crd();
47 au.CartesianToCell(c);
48 xa.ccrd() = c;
49 xa.CAtom().ccrd() = c;
50 }
51 }
52 olxstr ReCon; // restraint or constraint to use for split atoms
53 bool doSplit;
54public:
55 TSplitMode(size_t id) : AMode(id) {
56 gxapp.OnObjectsCreate.Add(this, mode_split_ObjectsCreate, msiExit);
57 gxapp.XFile().GetLattice().OnDisassemble.Add(
58 this, mode_split_Disassemble, msiEnter);
59 }

61 bool Initialise_(TStrObjList& Cmds, const TParamList& Options) {
62 if (!gxapp.CheckFileType<TIns>()) return false;
63 ReCon = Options.FindValue("r", EmptyString()).ToLowerCase();
64 olex2.processMacro("cursor(hand)");
65 TGXApp::AtomIterator ai = gxapp.GetAtoms();
66 while (ai.HasNext()) {
67 ai.Next().SetMoveable(true);
68 }
69 gxapp.SetZoomAfterModelBuilt(gxapp
70 .GetOptions().GetBoolOption("model.center_on_update", true, true));
71 doSplit = Options.GetBoolOption('s', false, true);
72 return true;
73 }
74
75 ~TSplitMode() {
76 gxapp.OnObjectsCreate.Remove(this);
77 gxapp.XFile().GetLattice().OnDisassemble.Remove(this);
78 }
79
80 void Finalise_() {
81 RefinementModel& rm = gxapp.XFile().GetRM();
82 UpdateCrds();
83 // if this is not done here it interferes and may cause a crash
84 gxapp.XFile().GetLattice().OnDisassemble.Remove(this);
85 gxapp.OnObjectsCreate.Remove(this);
86 TGXApp::AtomIterator ai = gxapp.GetAtoms();
87 while (ai.HasNext()) {
88 ai.Next().SetMoveable(false);
89 }
90 if (SplitAtoms.IsEmpty() || !doSplit) {
91 gxapp.XFile().GetLattice().UpdateConnectivity();
92 return;
93 }
94 TCAtomPList to_isot;
95 XVar& xv = rm.Vars.NewVar(0.75);
96 for (size_t i = 0; i < SplitAtoms.Count(); i++) {
97 TCAtom& a = *SplitAtoms[i].GetA();
98 TCAtom& b = *SplitAtoms[i].GetB();
99 to_isot.Add(a);
100 const double sp = 1. / a.GetDegeneracy();
101 rm.Vars.AddVarRef(xv, a, catom_var_name_Sof, relation_AsVar, sp);
102 rm.Vars.AddVarRef(xv, b, catom_var_name_Sof, relation_AsOneMinusVar, sp);
103 int part = a.GetPart();
104 if (part == 0) {
105 part++;
106 }
107 a.SetPart(part);
108 a.SetOccu(0.75*sp);
109 b.SetPart(olx_sign(part)*(olx_abs(part) + 1));
110 b.SetOccu(0.25*sp);
111 TSimpleRestraint* sr = 0;
112 if (ReCon.IsEmpty());
113 else if (ReCon == "eadp") {
114 sr = &rm.rEADP.AddNew();
115 }
116 else if (ReCon == "isor") {
117 sr = &rm.rISOR.AddNew();
118 }
119 else if (ReCon == "simu") {
120 sr = &rm.rSIMU.AddNew();
121 }
122 if (sr != 0) {
123 sr->AddAtomPair(a, 0, b, 0);
124 }
125 }
126 gxapp.XFile().GetLattice().SetAnis(to_isot, false);
127 gxapp.XFile().GetLattice().Uniq();
128 }

130 virtual bool Dispatch(int msg, short id, const IOlxObject* Sender,
131 const IOlxObject* Data, TActionQueue *)
132 {
133 if (msg == mode_split_ObjectsCreate) {
134 TGXApp::AtomIterator ai = gxapp.GetAtoms();
135 while (ai.HasNext()) {
136 ai.Next().SetMoveable(true);
137 }
138 }
139 else if (msg == mode_split_Disassemble) {
140 UpdateCrds();
141 }
142 return true;
143 }
144
145 virtual bool OnObject_(AGDrawObject &obj) {
146 if (obj.Is<TXAtom>()) {
147 TXAtom *XA = &(TXAtom&)obj;
148 bool split = true;
149 for (size_t i = 0; i < SplitAtoms.Count(); i++)
150 if (*SplitAtoms[i].GetA() == XA->CAtom() ||
151 *SplitAtoms[i].GetB() == XA->CAtom())
152 {
153 split = false;
154 break;
155 }
156 if (split && doSplit) {
157 TXAtom &xa = gxapp.AddAtom(XA);
158 const TAsymmUnit& au = gxapp.XFile().GetAsymmUnit();
159 TAsymmUnit::TLabelChecker lck(au);
160 xa.SetMoveable(true);
161 xa.SetRoteable(true);
162 SplitAtoms.AddNew(&XA->CAtom(), &xa.CAtom());
163 int part = XA->CAtom().GetPart();
164 if (part == 0) {
165 part++;
166 }
167 XA->CAtom().SetPart(part);
168 xa.CAtom().SetPart(part + 1);
169 xa.crd() += 0.5;
170 vec3d c = au.Fractionalise(xa.crd());
171 xa.CAtom().ccrd() = c;
172 xa.ccrd() = c;
173 olxstr new_l = XA->GetLabel();
174 olxch lc = olxstr::o_tolower(new_l.GetLast());
175 if (olxstr::o_isalpha(lc) && new_l.Length() > 1 && lc < 'z') {
176 new_l[new_l.Length() - 1] = ++lc;
177 }
178 else {
179 new_l << 'a';
180 }
181 xa.CAtom().SetLabel(lck.CheckLabel(xa.CAtom(), new_l, 0, true), true);
182 if (xa.GetType() == iQPeakZ) {
183 xa.CAtom().SetQPeak(1.0);
184 }
185 gxapp.XFile().GetLattice().UpdateConnectivity();
186 gxapp.UpdateDuplicateLabels();
187 if (TXApp::DoUseSafeAfix()) {
188 gxapp.GetUndo().Push(
189 gxapp.XFile().GetLattice().ValidateHGroups(true, true));
190 }
191 }
192 else { // do selection then
193 UpdateSelectionCrds();
194 gxapp.GetRenderer().Select(*XA);
195 }
196 return true;
197 }
198 return false;
199 }
