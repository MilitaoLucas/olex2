19struct std_sorter {
20 template <class item_t, class comparator_t>
21 struct idx_value {
22 item_t item;
23 const comparator_t* cmp;
24 size_t index;
25
26 idx_value() {}
27 idx_value(const item_t& item, const comparator_t& cmp, size_t index)
28 : item(item), cmp(&cmp), index(index)
29 {}
30 bool operator < (const idx_value& other) const {
31 return cmp->Compare(this->item, other.item) < 0;
32 }
33 };

35 template <bool stable, class list_t, class comparator_t, class listener_t>
36 struct StdSorter_ {
37 StdSorter_(list_t& list_, const comparator_t& cmp_,
38 const listener_t& listener_)
39 : list(list_), cmp(cmp_), listener(listener_)
40 {}
41 void Sort() {
42 size_t lc = list.list.Count();
43 if (lc < 2) {
44 return;
45 }
46 typedef typename list_t::InternalAccessor::list_item_type item_t;
47 typedef idx_value<item_t, comparator_t> idx_value_t;
48 TArrayList<idx_value_t> data(lc);
49 for (size_t i = 0; i < lc; i++) {
50 data[i] = idx_value_t(list[i], cmp, i);
51 }
52 idx_value_t* last = &data.GetLast();
53 if (stable) {
54 std::stable_sort(&data[0], ++last);
55 }
56 else {
57 std::sort(&data[0], ++last);
58 }
59 //https://medium.com/@kevingxyz/permutation-in-place-8528581a5553
60 for (size_t i = 0; i < lc; i++) {
61 size_t tsw = data[i].index;
62 while (tsw < i) {
63 tsw = data[tsw].index;
64 }
65 list.list.Swap(i, tsw);
66 listener.OnSwap(i, tsw);
67 }
68 }
69 protected:
70 typename list_t::InternalAccessor list;
71 const comparator_t& cmp;
72 const listener_t& listener;
73 };
74 template <bool stable, class list_t, class comparator_t, class listener_t>
75 static StdSorter_<stable, list_t, comparator_t, listener_t>
76 Make(list_t& list, const comparator_t& cmp, const listener_t& listener) {
77 return StdSorter_<stable, list_t, comparator_t, listener_t>(
78 list, cmp, listener);
79 }
80};
