40 TSocketFS(const TUrl& url, int _max_attempts = 100) :
41 THttpFileSystem(url), max_attempts(_max_attempts), BaseValid(false)
42 {
43 if (Base().IsEmpty())
44 Base() = TBasicApp::GetInstanceDir() + ".cds/";
45 if (UseLocalFS()) {
46 try {
47 if (!TEFile::Exists(Base())) {
48 if (TEFile::MakeDir(Base())) {
49 BaseValid = true;
50 }
51 }
52 else {
53 BaseValid = true;
54 }
55 }
56 catch (...) {}
57 }
58 if (max_attempts < 0) {
59 max_attempts = 0;
60 }
61 else if (max_attempts > 32000) {
62 max_attempts = 32000;
63 }
64 }
65 virtual ~TSocketFS() {}
66 static bool CanUseLocalFS() { return UseLocalFS(); }
67 // allows creating temporary files in basedir/.cds/
68 static void SetUseLocalFS(bool v) { UseLocalFS() = v; }
69 /* sets the dir in which temporary files can be stores and enables
70 the use of it as in a call to SetUseLocalFS(true). If the folder
71 does not exists an attempt to create one will be made and in the case
72 of an error false will be return, tru return otherwise */
73 static bool InitLocalFSBase(const olxstr& base) {
74 if (!TEFile::Exists(base)) {
75 if (!TEFile::MakeDir(base)) {
76 return false;
77 }
78 }
79 Base() = base;
80 SetUseLocalFS(true);
81 return false;
82 }
