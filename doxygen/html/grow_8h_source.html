10#ifndef __OLX_GROW_MODE_H
11#define __OLX_GROW_MODE_H
12#include "xgrowline.h"

15class TGrowMode : public AMode {
16protected:
17 bool GrowShells;
18 short mode, part;
19 // the size is used to detrmine if the AU is being rebuilt
20 TEBitArray detached;
21public:
22 TGrowMode(size_t id)
23 : AMode(id), GrowShells(false), mode(0), part(0)
24 {}
25 bool Initialise_(TStrObjList& Cmds, const TParamList& Options) {
26 bool SI = Options.Contains('s'),
27 Cov = Options.Contains('c'),
28 VdW = Options.Contains('v'),
29 Rad = Options.Contains('r');
30 GrowShells = Options.Contains("shells");
31 mode = 0;
32 if (SI) mode = gmSInteractions;
33 if (Cov) mode = gmCovalent;
34 else if (VdW) {
35 mode = gmVanDerWaals;
36 olxstr vr = Options.FindValue('v');
37 gxapp.SetDeltaV(vr.IsEmpty() ? 2.0 : vr.ToDouble());
38 }
39 else if (Rad) {
40 mode = gmSameAtoms;
41 }
42 if (mode == 0) {
43 mode = gmCovalent;
44 }
45 if (Options.Contains('p')) {
46 if (mode != gmCovalent || Options.Contains('a')) {
47 TBasicApp::NewLogEntry(logError) << "Incompatible values";
48 return false;
49 }
50 olxstr pv = Options.FindValue('p');
51 part = pv.IsEmpty() ? -1 : pv.ToInt();
52 GrowShells = true;
53 }
54 // the AU rebuilding mode, enfoces grow shells and covalent bonds only
55 if (Options.Contains('a')) {
56 TAsymmUnit &au = gxapp.XFile().GetAsymmUnit();
57 detached.SetSize(au.AtomCount());
58 for (size_t i = 0; i < au.AtomCount(); i++) {
59 TCAtom &a = au.GetAtom(i);
60 if (!a.IsDetached() && a.GetType() == iQPeakZ) {
61 a.SetDetached(true);
62 detached.SetTrue(i);
63 }
64 }
65 gxapp.UpdateConnectivity();
66 GrowShells = true;
67 mode = gmCovalent;
68 }
69 olex2.processMacro("cursor(hand)");
70 gxapp.SetGrowMode(mode, Cmds);
71 gxapp.SetXGrowLinesVisible(true);
72 gxapp.SetZoomAfterModelBuilt(false);
73 if (Options.GetBoolOption('l')) {
74 TPtrList<TXGrowLine> gl = gxapp.GetGrowLines();
75 QuickSorter::Sort(gl,
76 ComplexComparator::Make(
77 FunctionAccessor::MakeConst(&TXGrowLine::Length),
78 TPrimitiveComparator()));
79 TTable tab(gl.Count(), 4);
80 tab.ColName(0) = "From";
81 tab.ColName(1) = "To";
82 tab.ColName(2) = "Symm";
83 tab.ColName(3) = "D";
84 for (size_t i = 0; i < gl.Count(); i++) {
85 tab[i][0] = gl[i]->XAtom().CAtom().GetResiLabel();
86 tab[i][1] = gl[i]->CAtom().GetResiLabel();
87 if (gl[i]->GetTransform().IsFirst()) {
88 tab[i][2] = 'I';
89 }
90 else {
91 tab[i][2] = TSymmParser::MatrixToSymmEx(gl[i]->GetTransform());
92 }
93 tab[i][3] = olxstr::FormatFloat(3, gl[i]->Length());
94 }
95 TBasicApp::NewLogEntry() << tab.CreateTXTList("Contacts", true, false, " ");
96 }
97 return true;
98 }
99 void Finalise_() {
100 gxapp.SetXGrowLinesVisible(false);
101 gxapp.SetZoomAfterModelBuilt(true);
102 if (!detached.IsEmpty()) {
103 TAsymmUnit &au = gxapp.XFile().GetAsymmUnit();
104 for (size_t i = 0; i < au.AtomCount(); i++) {
105 if (detached[i]) {
106 au.GetAtom(i).SetDetached(false);
107 }
108 }
109 gxapp.XFile().GetLattice().CompaqQ();
110 }
111 if (part != 0) {
112 TLattice& latt = gxapp.XFile().GetLattice();
113 LabelCorrector lc(latt.GetAsymmUnit(), TXApp::GetMaxLabelLength(),
114 TXApp::DoRenameParts());
115 for (size_t i = 0; i < latt.GetObjects().atoms.Count(); i++) {
116 TSAtom &a = latt.GetObjects().atoms[i];
117 a.CAtom().SetOccu(a.CAtom().GetOccu() / 2);
118 if (!a.GetMatrix().IsFirst()) {
119 TCAtom &ca = latt.GetAsymmUnit().NewAtom();
120 ca.ccrd() = a.ccrd();
121 ca.SetType(a.GetType());
122 ca.SetPart((int8_t)part);
123 ca.SetOccu(a.CAtom().GetOccu());
124 ca.SetLabel(a.GetLabel(), false);
125 lc.Correct(ca);
126 gxapp.XFile().GetRM().Vars.FixParam(ca, catom_var_name_Sof);
127 }
128 }
129 latt.Init();
130 }
131 }
132 void UpdateAU(size_t ac) {
133 if (detached.IsEmpty()) return;
134 TLattice& latt = gxapp.XFile().GetLattice();
135 for (size_t i = ac; i < latt.GetObjects().atoms.Count(); i++) {
136 TSAtom &aa = latt.GetObjects().atoms[i];
137 for (size_t j = 0; j < ac; j++) {
138 TSAtom &ab = latt.GetObjects().atoms[j];
139 if (aa.CAtom().GetId() == ab.CAtom().GetId()) {
140 ab.SetDeleted(true);
141 }
142 }
143 }
144 if (ac != latt.GetObjects().atoms.Count()) {
145 latt.UpdateAsymmUnit();
146 latt.Init();
147 }
148 }
149 virtual bool OnObject_(AGDrawObject& obj) {
150 TLattice& latt = gxapp.XFile().GetLattice();
151 if (obj.Is<TXGrowLine>()) {
152 TXGrowLine& xl = (TXGrowLine&)obj;
153 if (GrowShells && (mode == gmCovalent || mode == gmVanDerWaals)) {
154 size_t ac = latt.GetObjects().atoms.Count();
155 latt.GrowAtom(xl.CAtom(), xl.GetTransform());
156 UpdateAU(ac);
157 }
158 else {
159 latt.GrowFragment(xl.CAtom().GetFragmentId(), xl.GetTransform());
160 }
161 return true;
162 }
163 else if (obj.Is<TXAtom>()) {
164 size_t ac = latt.GetObjects().atoms.Count();
165 TXAtom& a = (TXAtom&)obj;
166 if (!a.IsGrown()) {
167 latt.GrowAtom(a, GrowShells, NULL);
168 UpdateAU(ac);
169 }
170 return true;
171 }
172 else {
173 return false;
174 }
175 }
176};
