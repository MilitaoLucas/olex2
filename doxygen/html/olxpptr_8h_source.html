14/* a virtual pointer implementation for perishable objects - objects which can
15be deleted before this pointer container. Use is_valid() to check if the
16underlying object exists
17*/
18template <typename ptr> struct olx_perishable_ptr
19 : public olx_object_ptr<olx_virtual_ptr<ptr> >, public virtual IOlxObject
20{
21 typedef olx_object_ptr<olx_virtual_ptr<ptr> > parent_t;
22 typedef olx_vptr<ptr> vptr_t;
23
24 void on_object_delete(APerishable* o) {
25 parent_t::operator = (0);
26 }
27 void add_handler(APerishable& o) {
28 o.AddDestructionObserver(DestructionObserver::MakeNew(this,
29 &olx_perishable_ptr::on_object_delete));
30 }
31 void remove_handler() {
32 if (ok()) {
33 parent_t::p->p->get().RemoveDestructionObserver(
34 DestructionObserver::Make(this, &olx_perishable_ptr::on_object_delete));
35 }
36 }
37 olx_perishable_ptr(ptr *_p)
38 : parent_t(new typename vptr_t::template actual_ptr<ptr>(_p))
39 {
40 if (_p != 0) {
41 add_handler(*_p);
42 }
43 }
44 olx_perishable_ptr(olx_virtual_ptr<ptr> *_p)
45 : parent_t(_p)
46 {
47 if (_p->get_ptr() != 0) {
48 add_handler(_p->get());
49 }
50 }
51 template <class dptr>
52 olx_perishable_ptr(olx_virtual_ptr<dptr> *_p)
53 : parent_t(new typename vptr_t:: template ptr_proxy<dptr>(_p))
54 {
55 if (_p->get_ptr() != 0) {
56 add_handler(_p->get());
57 }
58 }
59 olx_perishable_ptr(const olx_perishable_ptr& _p)
60 : parent_t(_p)
61 {}
62 virtual ~olx_perishable_ptr() {
63 remove_handler();
64 }
65 olx_perishable_ptr& operator = (const olx_perishable_ptr& _p) {
66 if (_p.self_ptr() == this) {
67 return *this;
68 }
69 if (parent_t::p->template dec_ref<true, false>() != 0) {
70 remove_handler();
71 }
72 parent_t::p = _p.p->inc_ref();
73 if (ok()) {
74 add_handler(parent_t::p->p->get());
75 }
76 return *this;
77 }
78 olx_perishable_ptr& operator = (ptr *_p) {
79 remove_handler();
80 parent_t::operator = (new typename vptr_t:: template actual_ptr<ptr>(_p));
81 if (ok()) {
82 add_handler(parent_t::p->p->get());
83 }
84 return *this;
85 }
86
87 bool ok() const {
88 return (parent_t::ok() && parent_t::p->p->get_ptr() != 0);
89 }
90
91 const ptr* operator ->() const { return &parent_t::p->p->get(); }
92 ptr* operator ->() { return &parent_t::p->p->get(); }
93
94 const ptr* operator &() const { return &parent_t::p->p->get(); }
95 ptr* operator &() { return &parent_t::p->p->get(); }
96
97 const ptr& operator *() const { return parent_t::p->p->get(); }
98 ptr& operator *() { return parent_t::p->p->get(); }
99
100 operator const ptr& () const { return parent_t::p->p->get(); }
101 operator ptr& () { return parent_t::p->p->get(); }
102
103 bool operator == (const olx_perishable_ptr &ap) const {
104 return parent_t::p->p->get_ptr() == ap.p->p->get_ptr();
105 }
106 bool operator == (const ptr *p) const {
107 return parent_t::p->p->get_ptr() == p;
108 }
109 // releases the object from ALL references
110 ptr *release() {
111 if (!ok()) {
112 return 0;
113 }
114 ptr &p_ = parent_t::p->p->get();
115 remove_handler();
116 parent_t::operator = (0);
117 return &p_;
118 }
119protected:
120 const olx_perishable_ptr* self_ptr() const { return this; }
121};
