26 // if the atoms share a single residue -it is returned, 0 otherwise
27 template <class atom_plist_t, class accessor_t>
28 TResidue* get_resi(const atom_plist_t& atoms, const accessor_t& ac=DummyAccessor()) {
29 if (atoms.IsEmpty()) {
30 return 0;
31 }
32 size_t r_id = ac(atoms[0]).GetResiId();
33 for (size_t i = 1; i < atoms.Count(); i++) {
34 if (ac(atoms[i]).GetResiId() != r_id) {
35 return 0;
36 }
37 }
38 return &ac(atoms[0]).GetParent()->GetResidue(r_id);
39 }

42 olxstr label(const atom_plist_t &atoms, bool add_sym, const olxstr &sp) {
43 olxstr_buf rv;
44 for( size_t i=0; i < atoms.Count(); i++ ) {
45 rv << sp << (add_sym ? atoms[i]->GetGuiLabelEx() : atoms[i]->GetGuiLabel());
46 }
47 return rv.IsEmpty() ? EmptyString() : olxstr(rv).SubStringFrom(sp.Length());
48 }

53 return true;

61 if (!atoms[i]->GetMatrix().IsFirst()) {
62 return false;
63 }
64 }
65 return true;
