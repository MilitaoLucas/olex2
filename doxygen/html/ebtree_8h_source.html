15template <typename C, typename O> class BTree {
16public:
17 struct Entry {
18 C key;
19 O val;
20 Entry* next, *last; // linked list of items with equal key
21 Entry* left, *right;
22 Entry() {
23 next = last = left = right = NULL;
24 }
25 Entry(const C& key, const O& val) {
26 this->key = key;
27 this->val = val;
28 next = last = left = right = NULL;
29 }
30 ~Entry() { Clear(); }
31 inline void Clear() {
32 if( next ) delete next;
33 next = NULL;
34 if( right ) {
35 right->Clear();
36 delete right;
37 right = NULL;
38 }
39 if( left ) {
40 left->Clear();
41 delete left;
42 left = NULL;
43 }
44 }
45 };
46protected:
47 Entry Root;
48 int _Count;
49public:
50 BTree() : _Count(0) { }
51
52 ~BTree() { Root.Clear(); }
53
54 void Clear() { Root.Clear(); _Count = 0; }
55
56 inline const Entry* GetRoot() const { return (_Count == 0) ? NULL : &Root; }
57 inline int Count() const { return _Count; }
58
59 O& Add(const C& key, const O& val, Entry* insertPoint=NULL) {
60 Entry* rv = NULL;
61 if( _Count == 0 ) {
62 Root.key = key;
63 Root.val = val;
64 rv = &Root;
65 }
66 else {
67 Entry* en = (insertPoint == NULL) ? &Root : insertPoint;
68 while( en->key < key ) {
69 if( en->right && en->right->key <= key )
70 en = en->right;
71 else
72 break;
73 }
74 while( en->key > key ) {
75 if( en->left && en->left->key >= key )
76 en = en->left;
77 else
78 break;
79 }
80 if( en->key == key ) {
81 if( en->last ) {
82 rv = en->last->next = new Entry(key, val);
83 en->last = en->last->next;
84 }
85 else
86 rv = en->last = en->next = new Entry(key, val);
87 }
88 else if( en->key < key )
89 rv = en->right = new Entry(key, val);
90 else if( en->key > key )
91 rv = en->left = new Entry(key, val);
92 }
93 _Count++;
94 return rv->val;
95 }
96 /* find an entry by the key, if entry is not found and closest is provided,
97 closest is initlised with the entry to accept the key for the next add operation */
98 O* Find(const C& key, Entry** insertPoint=NULL) {
99 if( _Count == 0 ) {
100 if( insertPoint ) *insertPoint = &Root;
101 return NULL;
102 }
103 Entry* en = &Root;
104 while( en->key < key ) {
105 if( en->right && en->right->key <= key )
106 en = en->right;
107 else
108 break;
109 }
110 while( en->key > key ) {
111 if( en->left && en->left->key >= key )
112 en = en->left;
113 else
114 break;
115 }
116 if( insertPoint != NULL ) *insertPoint = en;
117 return (en->key == key) ? &en->val : NULL;
118 }

120 static TBTreeTraverser< BTree<C,O> > Traverser;
121};
