18// I. Krivy and B. Gruber. Acta Cryst. (1976) A32, 297-298
19struct Niggli {
20 struct EpsilonComparator {
21 double epsilon;
22 EpsilonComparator(double eps = 1e-3) : epsilon(eps) {}
23 bool eq(double a, double b) {
24 return olx_abs(a - b) < epsilon;
25 }
26 bool lt(double a, double b) {
27 return a < b - epsilon;
28 }
29 bool gt(double a, double b) {
30 //return a > b + epsilon;
31 return b < a - epsilon;
32 }
33 };
34 template <typename arr_t>
35 static evecd::const_vec_type reduce_const(short _latt, const arr_t& arr) {
36 vec3d s(arr[0], arr[1], arr[2]), a(arr[3], arr[4], arr[5]);
37 reduce(_latt, s, a);
38 return evecd::build(s[0], s[1], s[2], a[0], a[1], a[2]).release();
39 }
40 template <typename arr_t>
41 static arr_t& reduce(short _latt, arr_t& arr) {
42 vec3d s(arr[0], arr[1], arr[2]), a(arr[3], arr[4], arr[5]);
43 reduce(_latt, s, a);
44 arr[0] = s[0]; arr[1] = s[1]; arr[2] = s[2];
45 arr[3] = a[0]; arr[4] = a[1]; arr[5] = a[2];
46 return arr;
47 }
48 static void reduce(short _latt, vec3d& sides, vec3d& angles) {
49 double R[6];
50 vec3d cs(cos(angles[0] / 180.0 * M_PI), cos(angles[1] / 180.0 * M_PI),
51 cos(angles[2] / 180.0 * M_PI));
52 double vol = sides.Prod() * sqrt(1 - cs.QLength() + 2 * cs.Prod());
53 // Acta Cryst. (2004). A60, 1–6
54 EpsilonComparator cmp(1e-5 * pow(vol, 1. / 3));
55
56 mat3d metr_m(
57 olx_sqr(sides[0]), sides[0] * sides[1] * cs[2], sides[0] * sides[2] * cs[1],
58 olx_sqr(sides[1]), sides[1] * sides[2] * cs[0],
59 olx_sqr(sides[2]));
60
61 mat3d tm = TCLattice::FromPrimitive(_latt);
62 tm = tm.Inverse();
63 metr_m = tm.GetT() * metr_m * tm;
64
65 R[0] = metr_m[0][0];
66 R[1] = metr_m[1][1];
67 R[2] = metr_m[2][2];
68
69 R[3] = 2 * metr_m[1][2];
70 R[4] = 2 * metr_m[0][2];
71 R[5] = 2 * metr_m[0][1];
72 bool Used = true;
73 int Cycles = 0;
74 while (Used) {
75 Used = false;
76 Cycles++;
77 if (Cycles > 100) {
78 throw TFunctionFailedException(__OlxSourceInfo, "procedure did not converge");
79 }
80 if (cmp.gt(R[0], R[1]) ||
81 (cmp.eq(R[0], R[1]) && cmp.gt(olx_abs(R[3]), olx_abs(R[4]))))
82 { //1
83 olx_swap(R[0], R[1]);
84 olx_swap(R[3], R[4]);
85 Used = true;
86 }
87 if (cmp.gt(R[1], R[2]) ||
88 (cmp.eq(R[1], R[2]) && cmp.gt(olx_abs(R[4]), olx_abs(R[5]))))
89 { //2
90 olx_swap(R[1], R[2]);
91 olx_swap(R[4], R[5]);
92 Used = true;
93 continue;
94 }
95 if (cmp.gt(R[3] * R[4] * R[5], 0)) { //3
96 R[3] = olx_abs(R[3]);
97 R[4] = olx_abs(R[4]);
98 R[5] = olx_abs(R[5]);
99 }
100 else {
101 R[3] = -olx_abs(R[3]);
102 R[4] = -olx_abs(R[4]);
103 R[5] = -olx_abs(R[5]);
104 }
105 if (cmp.gt(olx_abs(R[3]), R[1]) ||
106 (cmp.eq(R[3], R[1]) && cmp.lt(2 * R[4], R[5])) || //5
107 (cmp.eq(R[3], -R[1]) && cmp.lt(R[5], 0)))
108 {
109 double sig = olx_sign(R[3]);
110 R[2] += R[1] - R[3] * sig;
111 R[3] -= 2 * R[1] * sig;
112 R[4] -= R[5] * sig;
113 Used = true;
114 continue;
115 }
116 if (cmp.gt(olx_abs(R[4]), R[0]) ||
117 (cmp.eq(R[4], R[0]) && cmp.lt(2 * R[3], R[5])) || //6
118 (cmp.eq(R[4], -R[0]) && cmp.lt(R[5], 0)))
119 {
120 double sig = olx_sign(R[4]);
121 R[2] += R[0] - R[4] * sig;
122 R[3] -= R[5] * sig;
123 R[4] -= 2 * R[0] * sig;
124 Used = true;
125 continue;
126 }
127 if (cmp.gt(olx_abs(R[5]), R[0]) ||
128 (cmp.eq(R[5], R[0]) && cmp.lt(2 * R[3], R[4])) || //7
129 (cmp.eq(R[5], -R[0]) && cmp.lt(R[4], 0)))
130 {
131 double sig = olx_sign(R[5]);
132 R[1] += R[0] - R[5] * sig;
133 R[3] -= R[4] * sig;
134 R[5] -= 2 * R[0] * sig;
135 Used = true;
136 continue;
137 }
138 if (cmp.lt(R[0] + R[1] + R[3] + R[4] + R[5], 0) ||
139 (cmp.eq(R[0] + R[1] + R[3] + R[4] + R[5], 0) && cmp.gt(2 * (R[0] + R[4]) + R[5], 0))) //8
140 {
141 R[2] += R[0] + R[1] + R[3] + R[4] + R[5];
142 R[3] += 2 * R[1] + R[5];
143 R[4] += 2 * R[0] + R[5];
144 Used = true;
145 }
146 }
147 sides[0] = sqrt(R[0]);
148 sides[1] = sqrt(R[1]);
149 sides[2] = sqrt(R[2]);
150 angles[0] = acos(R[3] / (sides[1] * sides[2] * 2)) * 180 / M_PI;
151 angles[1] = acos(R[4] / (sides[0] * sides[2] * 2)) * 180 / M_PI;
152 angles[2] = acos(R[5] / (sides[0] * sides[1] * 2)) * 180 / M_PI;
153 }
154};
