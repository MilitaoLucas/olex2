42 static size_t GetLattMultiplier(short latt) {
43 size_t count = 0;
44 switch( olx_abs(latt) ) {
45 case 1: count = 1; break; // P
46 case 2: // Body Centered (I)
47 case 5: // A Centered (A)
48 case 6: // B Centered (B)
49 case 7: count = 2; break; // C Centered (C);
50 case 8: // S Centered
51 case 9: // T Centered
52 case 10: // D Centered
53 case 11: // H Centered
54 case 3: count = 3; break; // R Centered
55 case 4: count = 4; break; // Face Centered (F)
56 default:
57 throw TIncorrectLattExc(__OlxSourceInfo, latt);
58 }
59 if( latt > 0 ) count *= 2;
60 return count;
61 }

356 template <typename MatList> void ExpandLatt(smatd_list& out,
357 const MatList& ml, short _latt) const
358 {
359 const TCLattice& latt = GetLatticeByNumber(_latt);
360 out.SetCapacity(latt.GetVectors().Count() * ml.Count() * (_latt > 0 ? 2 : 1));
361 out.AddNew().r.I();
362 for (size_t i = 0; i < ml.Count(); i++) {
363 const smatd& m = ml[i];
364 if (!m.r.IsI()) { // skip I matrix - always the first one
365 out.AddNew(m);
366 }
367 }
368 const size_t mc = out.Count();
369 for (size_t i = 0; i < latt.GetVectors().Count(); i++) {
370 for (size_t j = 0; j < mc; j++) {
371 out.AddCopy(out[j]).t += latt.GetVectors()[i];
372 }
373 }
374 if (_latt > 0) {
375 for (size_t j = 0; j < mc; j++) out.AddCopy(out[j].Negate());
376 for (size_t i = 0; i < latt.GetVectors().Count(); i++) {
377 for (size_t j = 0; j < mc; j++) {
378 out.AddCopy(out[j].Negate()).t += latt.GetVectors()[i];
379 }
380 }
381 }
382 }
