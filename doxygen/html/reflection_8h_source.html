148 template <class VC, class MC>

150 return VC((hkl[0] * mat[0][0] + hkl[1] * mat[1][0] + hkl[2] * mat[2][0]),

153 }

182 the matrix index in it

184 template <class MatList> static vec3i Standardise(const vec3i& _hkl,
185 const MatList& ml, size_t* idx = 0)
186 {
187 if (idx != 0) {
188 *idx = 0;
189 }
190 vec3i new_hkl = _hkl;
191 for (size_t i = 0; i < ml.Count(); i++) {
192 vec3i hklv = _hkl * ml[i].r;
193 if ((hklv[2] > new_hkl[2]) ||
194 ((hklv[2] == new_hkl[2]) && (hklv[1] > new_hkl[1])) ||
195 ((hklv[2] == new_hkl[2]) && (hklv[1] == new_hkl[1]) &&
196 (hklv[0] > new_hkl[0])))
197 {
198 new_hkl = hklv;
199 if (idx != 0) {
200 *idx = i;
201 }
202 }
203 }
204 return new_hkl;
205 }

207 template <class MatList> static compd PhaseShift(const vec3i& hkl,
208 const MatList& ml, size_t idx)
209 {
210 if (idx > 0 && idx < ml.Count()) {
211 double ps = ml[idx].t.DotProd(hkl);
212 if (olx_abs(ps - olx_round(ps)) < 0.01) {
213 return 1;
214 }
215 return compd::polar(-2*M_PI * ps);
216 }
217 throw TInvalidArgumentException(__OlxSourceInfo, "index");
218 }

220 template <class MatList> static bool IsAbsent(const vec3i& hkl,
221 const MatList& ml)
222 {
223 for (size_t i = 0; i < ml.Count(); i++) {
224 vec3i hklv = hkl * ml[i].r;
225 if (hkl == hklv) { // only if there is no change
226 const double ps = ml[i].t.DotProd(hkl);
227 if (olx_abs(ps - olx_round(ps)) > 0.01) {
228 return true;
229 }
230 }
231 }
232 return false;
233 }

277 template <class MatList> void Analyse(const MatList& ml) {
278 _reset_flags(0, 1, GetBatch());
279 for (size_t i = 0; i < ml.Count(); i++) {
280 vec3i hklv = MulHkl(hkl, ml[i]);
281 if (EqHkl(hklv)) { // symmetric reflection
282 IncMultiplicity();
283 if (!IsAbsent()) {
284 const double l = PhaseShift(ml[i]);
285 SetAbsent(olx_abs(l - olx_round(l)) > 0.01);
286 }
287 }
288 else if (!IsCentric() && EqNegHkl(hklv)) { // centric reflection
289 SetCentric(true);
290 }
291 }
292 }

437// generates miller indices in given range
438class MillerIndexArray : public IMillerIndexList {
439 vec3i min_i, max_i;
440 size_t h_sz, hk_sz, hkl_sz;
441public:
442 MillerIndexArray(const vec3i& mi, const vec3i& mx) : min_i(mi), max_i(mx) {
443 h_sz = (mx[0]-mi[0]+1);
444 hk_sz = h_sz*(mx[1]-mi[1]+1);
445 hkl_sz = hk_sz*(mx[2]-mi[2]+1);
446 }
447 size_t Count() const { return hkl_sz; }
448 vec3i operator [] (size_t i) const {
449 size_t pi = i/hk_sz, ipi = i%hk_sz;
450 return vec3i(
451 min_i[0]+(int)(ipi%h_sz),
452 min_i[1]+(int)(ipi/h_sz),
453 min_i[2]+(int)pi);
454 }
455};

471 template <class list_t>
472 static olx_pair_t<double, double> FindIRange(const list_t &l) {
473 if (l.IsEmpty()) {
474 return olx_pair::make(0.0, 0.0);
475 }
476 olx_pair_t<double, double> m;
477 m.b = m.a = olx_ref::get(l[0]).GetI();
478 for (size_t i = 1; i < l.Count(); i++) {
479 TReflection &r = olx_ref::get(l[i]);
480 if (r.GetI() > m.b) {
481 m.b = r.GetI();
482 }
483 if (r.GetI() < m.a) {
484 m.a = r.GetI();
485 }
486 }
487 return m;
488 }

490 // considers only positive MAX value!
491 template <class list_t>
492 static double CalcScale(const list_t &l) {
493 olx_pair_t<double, double> m = FindIRange(l);
494 return (m.b > 99999 ? 99999 / m.b : 1);
495 }

497 template <class list_t>
498 static TRefList::const_list_type Convert(const list_t &l, int ct) {
499 TRefList rv;
500 for (size_t i = 0; i < l.Count(); i++) {
501 TReflection &r = olx_ref::get(l[i]);
502 TReflection &rc = rv.AddCopy(r);
503 if (ct == reflection_convert_F_to_Fsq) {
504 rc.SetI(TReflection::F_as_Fsq(r.GetI(), r.GetS()));
505 }
506 else if (ct == reflection_convert_Fsq_toF) {
507 rc.SetI(TReflection::Fsq_as_F(r.GetI(), r.GetS()));
508 }
509 }
510 return rv;
511 }
