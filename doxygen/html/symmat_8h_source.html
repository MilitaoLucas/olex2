186 // normalises a fractional coordinate/translation
187 static vec3d NormaliseFractional(const vec3d &t_) {
188 vec3d t = t_;
189 for (int j=0; j < 3; j++) {
190 t[j] -= olx_floor(t[j]);
191 if (t[j] < 1e-3 || t[j] > 0.999) t[j] = 0;
192 }
193 return t;
194 }
195 /* returns a vector with smaller or equal values */
196 template <class list_t, class accessor_t>
197 static vec3d StandardiseFractional(const vec3d &x, const list_t& ml,
198 const accessor_t &acc)
199 {
200 vec3d v = NormaliseFractional(x);
201 for( size_t i=0; i < ml.Count(); i++ ) {
202 vec3d tmp = NormaliseFractional(acc(ml[i])*x);
203 if( (tmp[0] < v[0]) || // standardise then ...
204 (olx_abs(tmp[0]-v[0]) < 1e-3 && (tmp[1] < v[1])) ||
205 (olx_abs(tmp[0]-v[0]) < 1e-3 &&
206 olx_abs(tmp[1]-v[1]) < 1e-3 && (tmp[2] < v[2])) )
207 {
208 v = tmp;
209 }
210 }
211 return v;
212 }
213 template <class list_t>
214 static vec3d StandardiseFractional(const vec3d &x, const list_t& ml) {
215 return StandardiseFractional(x, ml, ListAccessor(ml));
