96 template <class T> inline TCString& assignTypeValue(const char* format, T v) {
97 olx_array_ptr<char> bf(80);
98#if defined(_MSC_VER)
99 sprintf_s(*bf, 80, format, v);
100#else
101 sprintf(*bf, format, v);
102#endif
103 _Start = 0;
104 _Increment = 8;
105 _Length = strlen(bf);
106 if (SData != 0) {
107 if (SData->RefCnt == 1) { // owed by this object
108 SData->SetCapacity(_Length);
109 olx_memcpy(SData->Data, bf, _Length);
110 }
111 else {
112 SData->RefCnt--;
113 SData = 0;
114 }
115 }
116 if (SData == 0) {
117 SData = new Buffer(_Length + _Increment, bf, _Length);
118 }
119 return *this;
