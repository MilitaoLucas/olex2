16struct MergeStats {
17 double Rint, Rsigma, MeanIOverSigma;
18 size_t
19 SystematicAbsencesRemoved,
20 UniqueSystematicAbsencesRemoved,
21 InconsistentEquivalents,
22 UniqueReflections,
23 CentricReflections,
24 ReflectionAPotMax,
25 GTRefs, // I/sig >= 2
26 OmittedByUser; // OMIT h k l, all equivs
27 // symmetry independent reflections = sum(1/multiplicity)
28 double IndependentReflections;
29 bool FriedelOppositesMerged;
30 vec3i MinIndices, MaxIndices;
31 MergeStats() {
32 SetDefaults();
33 }
34 MergeStats(const MergeStats& ms) {
35 this->operator = (ms);
36 }
37 MergeStats& operator = (const MergeStats& ms) {
38 Rint = ms.Rint;
39 Rsigma = ms.Rsigma;
40 MeanIOverSigma = ms.MeanIOverSigma;
41 SystematicAbsencesRemoved = ms.SystematicAbsencesRemoved;
42 UniqueSystematicAbsencesRemoved = ms.UniqueSystematicAbsencesRemoved;
43 InconsistentEquivalents = ms.InconsistentEquivalents;
44 UniqueReflections = ms.UniqueReflections;
45 OmittedByUser = ms.OmittedByUser;
46 CentricReflections = ms.CentricReflections;
47 FriedelOppositesMerged = ms.FriedelOppositesMerged;
48 GTRefs = ms.GTRefs;
49 ReflectionAPotMax = ms.ReflectionAPotMax;
50 MinIndices = ms.MinIndices;
51 MaxIndices = ms.MaxIndices;
52 IndependentReflections = ms.IndependentReflections;
53 return *this;
54 }
55 void SetDefaults() {
56 Rint = Rsigma = 0;
57 MeanIOverSigma = 0;
58 MinIndices[0] = MinIndices[1] = MinIndices[2] = 100;
59 MaxIndices[0] = MaxIndices[1] = MaxIndices[2] = -100;
60 GTRefs = OmittedByUser = UniqueReflections = 0;
61 CentricReflections = SystematicAbsencesRemoved =
62 UniqueSystematicAbsencesRemoved = InconsistentEquivalents = 0;
63 ReflectionAPotMax = 0;
64 FriedelOppositesMerged = false;
65 IndependentReflections = 0;
66 }
67};

70 template <class RefListMerger, class Comparator>
71 static MergeStats _DoMerge(const SymmSpace::InfoEx& info_ex, TRefPList& refs,
72 const vec3i_list& omits, TRefList& output, const Comparator &cmp, double gt_v)
73 {
74 const size_t ref_cnt = refs.Count();
75 if (ref_cnt == 0) {
76 throw TInvalidArgumentException(__OlxSourceInfo,
77 "empty reflection list");
78 }
79 for (size_t i = 0; i < ref_cnt; i++) {
80 refs[i]->Standardise(info_ex);
81 }
82 MergeStats stats;
83 stats.FriedelOppositesMerged = info_ex.centrosymmetric;
84 /* standardise reflection indices according to provided list of symmetry
85 operators
86 */
87 QuickSorter::Sort(refs, cmp);
88 output.SetCapacity(ref_cnt);
89 // merge reflections
90 double Sdiff = 0, SI = 0, SS = 0, SI_tot = 0;
91 size_t mi_o_sig_cnt = 0;
92 for (size_t i = 0; i < ref_cnt;) {
93 TReflection* ref = refs[i];
94 const size_t from = i++;
95 while ((i < ref_cnt) && (cmp.Compare(ref, refs[i]) == 0)) {
96 i++;
97 }
98 const size_t merged_count = i - from;
99 if (omits.Contains(ref->GetHkl())) {
100 stats.OmittedByUser += merged_count;
101 }
102 else {
103 if (merged_count > stats.ReflectionAPotMax) {
104 stats.ReflectionAPotMax = merged_count;
105 }
106 if (!ref->IsAbsent()) {
107 // the reflection is replicated if only one in the list
108 MergerOut mo = RefListMerger::Merge(refs, from, i);
109 if (merged_count > 1) {
110 SI_tot += mo.sumI;
111 Sdiff += mo.sumDiff;
112 if (mo.sigInt > mo.ref->GetS()) {
113 if (mo.sigInt > 5 * mo.ref->GetS()) {
114 stats.InconsistentEquivalents++;
115 mo.ref->SetTag(-1); // mark as unusable
116 }
117 mo.ref->SetS(mo.sigInt);
118 }
119 mo.ref->SetBatch(ref->GetBatch());
120 }
121 output.Add(mo.ref).Analyse(info_ex);
122 if (mo.ref->IsCentric()) {
123 stats.CentricReflections++;
124 }
125 SS += mo.ref->GetS();
126 SI += mo.ref->GetI();
127 if (mo.ref->GetS() > 0) {
128 double ios = mo.ref->GetI() / mo.ref->GetS();
129 if (ios >= gt_v) {
130 stats.GTRefs++;
131 }
132 stats.MeanIOverSigma += ios;
133 mi_o_sig_cnt++;
134 }
135 vec3i::UpdateMinMax(ref->GetHkl(), stats.MinIndices,
136 stats.MaxIndices);
137 stats.IndependentReflections += 1. / mo.ref->GetMultiplicity();
138 }
139 else {
140 stats.SystematicAbsencesRemoved += merged_count;
141 stats.UniqueSystematicAbsencesRemoved++;
142 }
143 }
144 }
145 stats.Rint = (SI_tot != 0) ? Sdiff / SI_tot : 0.0;
146 stats.Rsigma = (SI != 0) ? SS / SI : 0.0;
147 stats.UniqueReflections = output.Count();
148 if (mi_o_sig_cnt > 0) {
149 stats.MeanIOverSigma /= mi_o_sig_cnt;
150 }
151 return stats;
152 }

232 template <class MatList, class Comparator, bool UseBatch>
233 static MergeStats _DoDrySGFilter(const MatList& ml, TRefPList& refs,
234 const vec3i_list& omits, const Comparator &cmp, double gt_v)
235 {
236 const size_t ref_cnt = refs.Count();
237 if (ref_cnt == 0) {
238 throw TInvalidArgumentException(__OlxSourceInfo,
239 "empty reflection list");
240 }
241 MergeStats stats;
242 stats.FriedelOppositesMerged = false;
243 // sort the list
244 QuickSorter::Sort(refs, cmp);
245 size_t mi_o_sig_cnt = 0;
246 for (size_t i = 0; i < ref_cnt;) {
247 TReflection* ref = refs[i];
248 if (UseBatch && ref->GetBatch() < 0) {
249 continue;
250 }
251 ref->Analyse(ml);
252 size_t from = i++, real_count = 1;
253 while ((i < ref_cnt) && (cmp.Compare(ref, refs[i]) == 0)) {
254 if (!UseBatch || refs[i]->GetBatch() > 0) {
255 real_count++;
256 }
257 i++;
258 }
259 if (omits.Contains(ref->GetHkl())) {
260 stats.OmittedByUser += real_count;
261 }
262 else {
263 if (real_count > stats.ReflectionAPotMax) {
264 stats.ReflectionAPotMax = real_count;
265 }
266 if (!ref->IsAbsent()) {
267 for (size_t j = from; j < i; j++) {
268 vec3i::UpdateMinMax(refs[j]->GetHkl(), stats.MinIndices,
269 stats.MaxIndices);
270 if (!UseBatch || refs[j]->GetBatch() > 0) {
271 double ios = refs[j]->GetI() / refs[j]->GetS();
272 if (ios >= gt_v) {
273 stats.GTRefs++;
274 }
275 stats.MeanIOverSigma += ios;
276 mi_o_sig_cnt++;
277 }
278 }
279 if (ref->IsCentric()) {
280 stats.CentricReflections += real_count;
281 }
282 stats.UniqueReflections += real_count;
283 stats.IndependentReflections +=
284 (double)real_count / ref->GetMultiplicity();
285 }
286 else {
287 stats.SystematicAbsencesRemoved += real_count;
288 stats.UniqueSystematicAbsencesRemoved++;
289 }
290 }
291 }
292 stats.Rint = -1;
293 stats.Rsigma = -1;
294 if (mi_o_sig_cnt > 0) {
295 stats.MeanIOverSigma /= mi_o_sig_cnt;
296 }
297 return stats;
298 }

625 /* merges using statistical weights for the intensities
626 Imean = sum( I/Sig^2 )/sum( 1./Sig^2 )
627 http://www.crystal.chem.uu.nl/distr/mergehklf5/mergehklf5.html
628 we cannot use a single pass algorithm like:
629 double sum_wght = 0, sum_wght_i_sq = 0, sum_wght_i = 0, sum_i = 0;
630 for( size_t l=0; l < rl.Count(); l++ ) {
631 const double s = rl[l]->GetS() != 0 ? rl[l]->GetS() : 0.001;
632 const double w = 1./(s*s);
633 sum_wght += w;
634 sum_wght_i_sq += w*rl[l]->GetI()*rl[l]->GetI();
635 sum_wght_i += w*rl[l]->GetI();
636 sum_i += rl[l]->GetI();
637 }
638 const double mean = sum_wght_i/sum_wght;
