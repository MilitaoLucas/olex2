94 template <typename FT> TVector3<FT>& CellToCartesian(TVector3<FT>& crt) const
95 {
96 crt[0] = (FT)(crt[0] * Cell2Cartesian[0][0] + crt[1] * Cell2Cartesian[1][0] +
97 crt[2] * Cell2Cartesian[2][0]);
98 crt[1] = (FT)(crt[1] * Cell2Cartesian[1][1] + crt[2] * Cell2Cartesian[2][1]);
99 crt[2] = (FT)(crt[2] * Cell2Cartesian[2][2]);
100 return crt;
101 //crt *= Cell2Cartesian;
102 }

111 template <typename FT> TVector3<FT>& CartesianToCell(TVector3<FT>& cll) const

114 cll[2] * Cartesian2Cell[2][0]);
115 cll[1] = (FT)(cll[1] * Cartesian2Cell[1][1] + cll[2] * Cartesian2Cell[2][1]);
116 cll[2] = (FT)(cll[2] * Cartesian2Cell[2][2]);
117 return cll;

129 template <class T> T& UcifToUcart(T& v) const {
130 mat3d M(v[0], v[5], v[4], v[1], v[3], v[2]);
131 M = UcifToUxyz * M * UcifToUxyzT;
132 v[0] = M[0][0]; v[1] = M[1][1]; v[2] = M[2][2];
133 v[3] = M[1][2]; v[4] = M[0][2]; v[5] = M[0][1];
134 return v;
135 }

139 v[0] = M[0][0]; v[1] = M[1][1]; v[2] = M[2][2];
140 v[3] = M[1][2]; v[4] = M[0][2]; v[5] = M[0][1];
141 return v;
142 }
