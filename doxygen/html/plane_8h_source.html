17 // sorts points in plane given by center and normal
18 template <class array_t, class accessor_t, class vec_t>
19 void Sort(array_t &points, const accessor_t &accessor,
20 const vec_t &center, const vec_t &normal)
21 {
22 if( points.IsEmpty() ) return;
23 TArrayList<double> pvs(points.Count());
24 double max_qd = 0;
25 size_t m_i = 0;
26 for (size_t i=0; i < points.Count(); i++) {
27 double qd = (accessor(points[i])-center).QLength();
28 if (qd > max_qd) {
29 m_i = i;
30 max_qd = qd;
31 }
32 }
33 const vec_t origin = accessor(points[m_i])-center;
34 pvs[m_i] = 0; // origin value
35 for( size_t i=0; i < points.Count(); i++ ) {
36 if (i == m_i) {
37 continue;
38 }
39 vec_t vec = accessor(points[i]) - center;
40 if (vec.QLength() < 1e-6) { // atom on the center!
41 continue;
42 }
43 double ca = origin.CAngle(vec);
44 vec = origin.XProdVec(vec);
45 double vo;
46 if (vec.IsNull(1e-6)) { // same direction as the origin
47 vo = 0;
48 }
49 else {
50 vo = (ca == -1 ? 0 : vec.CAngle(normal));
51 }
52 /* negative - vec is on the right, positive - on the left, if ca == 0,
53 vec == (0,0,0)
54 */
55 if( ca >= 0 ) { // -90 to 90
56 if (vo < 0) { // -90 to 0 3->4
57 pvs[i] = 3.0 + ca;
58 }
59 else { // 0 to 90 0->1
60 pvs[i] = 1.0 - ca;
61 }
62 }
63 else if( ca > -1 ) { // 90-270
64 if (vo < 0) { // 180 to 270 2->3
65 pvs[i] = 3.0 + ca;
66 }
67 else { // 90 to 180 1->2
68 pvs[i] = 1.0 - ca;
69 }
70 }
71 else { //-1, special case
72 pvs[i] = 2;
73 }
74 }
75 QuickSorter::Sort(pvs, TPrimitiveComparator(),
76 SyncSortListener::MakeSingle(points));
77 }

111 template <class list_t, class p_accessor_t, class w_accessor_t>
112 static out calc(const list_t& list,
113 const p_accessor_t& p_accessor,
114 const w_accessor_t& w_accessor)
115 {
116 out rv;
117 if (list.Count() < 2) {
118 return rv;
119 }
120 FloatT w_sum = 0, w_sq_sum = 0;
121 for (size_t i = 0; i < list.Count(); i++) {
122 FloatT w = w_accessor(list[i]);
123 rv.center += p_accessor(list[i]) * w;
124 w_sum += w;
125 w_sq_sum += w*w;
126 }
127 rv.center /= w_sum;
128 TMatrix33<FloatT> m;
129 for (size_t i = 0; i < list.Count(); i++) {
130 TVector3<FloatT> t =
131 (p_accessor(list[i]) - rv.center) * w_accessor(list[i]);
132 m[0][0] += (t[0] * t[0]);
133 m[0][1] += (t[0] * t[1]);
134 m[0][2] += (t[0] * t[2]);
135 m[1][1] += (t[1] * t[1]);
136 m[1][2] += (t[1] * t[2]);
137 m[2][2] += (t[2] * t[2]);
138 }
139 m[1][0] = m[0][1];
140 m[2][0] = m[0][2];
141 m[2][1] = m[1][2];
142 TMatrix33<FloatT>::EigenValues(m /= w_sq_sum, rv.normals.I());
143 for (int i = 0; i < 3; i++) {
144 rv.rms[i] = m[i][i] < 0 ? 0 : sqrt(m[i][i]);
145 }
146 rv.sort();
147 return rv;
148 }

169 template <typename FloatT=double>
170 struct center {
171 template <class list_t, class p_accessor_t, class w_accessor_t>
172 static TVector3<FloatT> calc(const list_t& list,
173 const p_accessor_t& p_accessor,
174 const w_accessor_t& w_accessor)
175 {
176 TVector3<FloatT> c;
177 if (list.Count() == 0) {
178 return c;
179 }
180 FloatT w_sum = 0;
181 for (size_t i = 0; i < list.Count(); i++) {
182 FloatT w = w_accessor(list[i]);
183 c += p_accessor(list[i]) * w;
184 w_sum += w;
185 }
186 c /= w_sum;
187 return c;
188 }
189 /* calculates a center of the object reflected in a plane defined by the
190 given normal
191 */
192 template <class list_t, class p_accessor_t, class w_accessor_t>
193 static TVector3<FloatT> calc_reflected(const list_t& list,
194 const TVector3<FloatT> &center,
195 const TVector3<FloatT> &normal,
196 const p_accessor_t& p_accessor,
197 const w_accessor_t& w_accessor)
198 {
199 TVector3<FloatT> c;
200 if (list.Count() == 0) {
201 return c;
202 }
203 FloatT w_sum = 0;
204 for (size_t i = 0; i < list.Count(); i++) {
205 FloatT w = w_accessor(list[i]);
206 c += ((p_accessor(list[i])-center).Reflect(normal) + center) * w;
207 w_sum += w;
208 }
209 c /= w_sum;
210 return c;
211 }
