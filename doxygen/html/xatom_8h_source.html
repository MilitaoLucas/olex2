134 double GetDrawScale() const {
135 double scale = FParams[1];
136 if ((FRadius & darIsot) != 0) {
137 if (GetType().z != 1) {
138 scale *= GetSettings().GetTelpProb();
139 }
140 }
141 else if ((FRadius & darIsotH) != 0) {
142 scale *= GetSettings().GetTelpProb();
143 }
144 if ((FDrawStyle == adsEllipsoid || FDrawStyle == adsOrtep) &&
145 GetEllipsoid() != NULL)
146 {
147 if (GetEllipsoid()->IsNPD()) {
148 return (caDefIso * 2 * scale);
149 }
150 return scale;
151 }
152 else {
153 return FParams[0] * scale;
154 }
