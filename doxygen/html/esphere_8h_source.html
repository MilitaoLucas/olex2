24// Octahedron face provider
25template <class vec_type> struct OctahedronFP {
26 static const vec_type *vertices() {
27 static vec_type v[6] = {
28 vec_type(0,0,1), vec_type(1,0,0), vec_type(0,1,0),
29 vec_type(-1,0,0), vec_type(0,-1,0), vec_type(0,0,-1)
30 };
31 return &v[0];
32 }
33 static const IndexTriangle *faces() {
34 static IndexTriangle f[8] = {
35 IndexTriangle(0, 1, 2), IndexTriangle(0, 2, 3),
36 IndexTriangle(0, 3, 4), IndexTriangle(0, 4, 1),
37 IndexTriangle(5, 2, 1), IndexTriangle(5, 3, 2),
38 IndexTriangle(5, 4, 3), IndexTriangle(5, 1, 4)
39 };
40 return &f[0];
41 }
42 static size_t vertex_count() { return 6; }
43 static size_t face_count() { return 8; }
44};
45// tetrahedron face provider
46// vertices are as from http://en.wikipedia.org/wiki/Tetrahedron
47template <class vec_type> struct TetrahedronFP {
48 static const vec_type *vertices() {
49 typedef typename vec_type::number_type FT;
50 const static FT k = (FT)1./sqrt((FT)3.0);
51 static vec_type v[4] = {
52 vec_type(k, k, k), vec_type(-k, -k, k), vec_type(-k, k, -k),
53 vec_type(k, -k, -k)
54 };
55 return &v[0];
56 }
57 static const IndexTriangle *faces() {
58 static IndexTriangle f[4] = {
59 IndexTriangle(0, 2, 1), IndexTriangle(0, 1, 3),
60 IndexTriangle(0, 3, 2), IndexTriangle(1, 2, 3)
61 };
62 return &f[0];
63 }
64 static size_t vertex_count() { return 4; }
65 static size_t face_count() { return 4; }
66};
67/* a class to build sphere by partitioning an octahedron. Alows custom masks to
68be aplied when rendering the sphere in OpenGL or other rendering engine. All
69methods and members can be made static...
70*/
71template <typename float_type, class FaceProvider> class OlxSphere {
72public:
73 /* rad - radius of the sphere, ext - the generation to generate, vo - vector
74 output, to - triangles output, normals - the normals output (one a vertex,
75 not triangle). Since there is a huge redundancy, a unique set of vecteces is
76 generated and triangles contain inly indeces to the verteces they are based
77 on
78 */
79 static void Generate(float_type rad, size_t ext,
80 TTypeList<TVector3<float_type> >& vo,
81 TTypeList<IndexTriangle>& to,
82 TArrayList<TVector3<float_type> >& normals)
83 {
84 Generate(rad, ext, vo, to, &normals);
85 }
86 static void Generate(float_type rad, size_t ext,
87 TTypeList<TVector3<float_type> >& vo,
88 TTypeList<IndexTriangle>& to,
89 TArrayList<TVector3<float_type> >* _normals=NULL)
90 {
91 size_t nc = FaceProvider::face_count();
92 for( size_t i=0; i < ext; i++ )
93 nc += nc*3;
94
95 to.SetCapacity(nc+1);
96 vo.SetCapacity(nc-1);
97 for( size_t i=0; i < FaceProvider::vertex_count(); i++ )
98 vo.AddNew(FaceProvider::vertices()[i]);
99 for( size_t i=0; i < FaceProvider::face_count(); i++ )
100 to.AddNew(FaceProvider::faces()[i]);
101 for( size_t i=0; i < ext; i++ ) {
102 const size_t t_cnt = to.Count();
103 for( size_t j=0; j < t_cnt; j++ ) {
104 IndexTriangle& t = to[j];
105 vo.AddNew(vo[t.vertices[0]]+vo[t.vertices[1]]).Normalise();
106 vo.AddNew(vo[t.vertices[0]]+vo[t.vertices[2]]).Normalise();
107 vo.AddNew(vo[t.vertices[1]]+vo[t.vertices[2]]).Normalise();
108 // new triangle, (12, 23, 13)
109 to.AddNew(vo.Count()-3, vo.Count()-1, vo.Count()-2);
110 // new tringles partially based on old vertices (new, 23, 12)
111 to.AddNew(t.vertices[1], vo.Count()-1, vo.Count()-3);
112 // new tringles partially based on old vertices (new, 13, 23)
113 to.AddNew(t.vertices[2], vo.Count()-2, vo.Count()-1);
114 // modify the original triangle to take two new points
115 t.vertices[1] = vo.Count()-3; // 12
116 t.vertices[2] = vo.Count()-2; // 13
117 }
118 }
119 if( _normals != NULL ) {
120 TArrayList<TVector3<float_type> >& normals = *_normals;
121 normals.SetCount(vo.Count());
122 for( size_t i=0; i < vo.Count(); i++ ) { // normalise the vertices
123 normals[i] = vo[i];
124 vo[i] *= rad;
125 }
126 }
127 else {
128 for( size_t i=0; i < vo.Count(); i++ )
129 vo[i] *= rad;
130 }
131 }
132};
