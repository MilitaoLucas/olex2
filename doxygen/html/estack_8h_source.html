18// the simplest stack implementation
19template <class T> class TStack {
20public:
21 struct item {
22 item* prev;
23 T data;
24 item(const T& v, item* _prev) : prev(_prev), data(v) {}
25 };
26private:
27 item* cur;
28 size_t _count;
29public:
30 TStack() : cur(NULL), _count(0) {}
31 virtual ~TStack() { Clear(); }
32 void Clear() {
33 while( cur != NULL ) {
34 item* p = cur->prev;
35 delete cur;
36 cur = p;
37 }
38 _count = 0;
39 }
40 inline T& Push(const T& v) {
41 item* ni = new item(v, cur);
42 cur = ni;
43 _count++;
44 return cur->data;
45 }
46 T Pop() {
47 if( cur != NULL ) {
48 item* i = cur->prev;
49 T rv = cur->data;
50 delete cur;
51 cur = i;
52 _count--;
53 return rv;
54 }
55 throw TFunctionFailedException(__OlxSourceInfo, "stack is empty");
56 }
57 bool IsEmpty() const { return cur == NULL; }
58 size_t Count() const { return _count; }
59 T& Top() {
60 if( cur == NULL )
61 throw TFunctionFailedException(__OlxSourceInfo, "stack is empty");
62 return cur->data;
63 }
64 // this for read-only traversal
65 const item* TopItem() const { return cur; }
66};

68class str_stack : public TStack<olxstr> {
69public:
70 str_stack(const olxstr& exp) { LoadFromExpression(exp); }
71 str_stack() {}

73 size_t LoadFromExpression(const olxstr& Exp) {
74 Clear();
75 int argc=0;
76 char op;
77 olxstr o;
78 bool NewOperation = false;
79 if( !Exp.IsEmpty() ) {
80 for( size_t i=0; i < Exp.Length(); i++ ) {
81 switch( Exp.CharAt(i) ) {
82 case '-': NewOperation = true; op = '-'; break;
83 case '+': NewOperation = true; op = '+'; break;
84 case ' ': break;
85 case '*': NewOperation = true; op = '*'; break;
86 case '/': NewOperation = true; op = '/'; break;
87 default:
88 if( NewOperation ) {
89 if( !o.IsEmpty() ) {
90 Push(o);
91 argc++;
92 }
93 Push(op);
94 o.SetLength(0);
95 NewOperation = false;
96 }
97 o << Exp[i];
98 break;
99 }
100 }
101 Push(o);
102 }
103 return argc;
104 }
105};
