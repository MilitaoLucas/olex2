17template <typename Float> struct Contour {
18 // feedback support structures
19 class IFeedback {
20 public:
21 virtual void OnLine(Float x1, Float y1, Float x2, Float y2, Float z) const = 0;
22 };
23 class StaticFeedback : public IFeedback {
24 protected:
25 void (*DrawLine)(Float,Float,Float,Float,Float);
26 public:
27 StaticFeedback(void (*_DrawLine)(Float,Float,Float,Float,Float)) : DrawLine(_DrawLine) {}
28 virtual void OnLine(Float x1, Float y1, Float x2, Float y2, Float z) const {
29 (*DrawLine)(x1, y1, x2, y2, z);
30 }
31 };
32
33 template <class Base> class MemberFeedback : public IFeedback {
34 protected:
35 Base& instance;
36 void (Base::*DrawLine)(Float,Float,Float,Float,Float);
37 public:
38 MemberFeedback(Base& _instance, void (Base::*_DrawLine)(Float,Float,Float,Float,Float)) :
39 instance(_instance), DrawLine(_DrawLine) {}
40 virtual void OnLine(Float x1, Float y1, Float x2, Float y2, Float z) const {
41 (instance.*DrawLine)(x1, y1, x2, y2, z);
42 }
43 };
44
45 template <class Base> class MemberFeedback<const Base&> : public IFeedback {
46 protected:
47 const Base& instance;
48 void (Base::*DrawLine)(Float,Float,Float,Float,Float) const;
49 public:
50 MemberFeedback(const Base& _instance, void (Base::*_DrawLine)(Float,Float,Float,Float,Float) const) :
51 instance(_instance), DrawLine(_DrawLine) {}
52 virtual void OnLine(Float x1, Float y1, Float x2, Float y2, Float z) const {
53 (instance.*DrawLine)(x1, y1, x2, y2, z);
54 }
55 };
56protected:
57 Float h[5];
58 int sh[5];
59 Float xh[5], yh[5];
60 inline Float xsect(size_t p1, size_t p2) const { return (h[p2]*xh[p1]-h[p1]*xh[p2])/(h[p2]-h[p1]); }
61 inline Float ysect(size_t p1, size_t p2) const { return (h[p2]*yh[p1]-h[p1]*yh[p2])/(h[p2]-h[p1]); }
62public:
63 void DoContour(Float **d, int x_start, int x_end, int y_start, int y_end,
64 Float *x, Float *y, size_t contour_number, Float *z,
65 IFeedback& callback)
66 {
67 static const size_t im[4] = {0,1,1,0}, jm[4]={0,0,1,1};
68 static const size_t castab[3][3][3] = {
69 { {0,0,8},{0,2,5},{7,6,9} },
70 { {0,3,4},{1,3,1},{4,3,0} },
71 { {9,6,7},{5,2,0},{8,0,0} }
72 };
73 for( int j=(y_end-1); j >= y_start; j-- ) {
74 for( int i=x_start; i <= x_end-1; i++) {
75 Float dmin = olx_min(olx_min(d[i][j],d[i][j+1]), olx_min(d[i+1][j],d[i+1][j+1]));
76 Float dmax = olx_max(olx_max(d[i][j],d[i][j+1]), olx_max(d[i+1][j],d[i+1][j+1]));
77 if (dmax < z[0] || dmin > z[contour_number-1])
78 continue;
79 for( size_t k=0; k < contour_number; k++ ) {
80 if (z[k] < dmin || z[k] > dmax)
81 continue;
82 for( int m=4; m >= 0; m-- ) {
83 if (m > 0) {
84 h[m] = d[i+im[m-1]][j+jm[m-1]]-z[k];
85 xh[m] = x[i+im[m-1]];
86 yh[m] = y[j+jm[m-1]];
87 } else {
88 h[0] = Float(0.25) * (h[1]+h[2]+h[3]+h[4]);
89 xh[0] = Float(0.50) * (x[i]+x[i+1]);
90 yh[0] = Float(0.50) * (y[j]+y[j+1]);
91 }
92 if( h[m] > 0 )
93 sh[m] = 1;
94 else if( h[m] < 0 )
95 sh[m] = -1;
96 else
97 sh[m] = 0;
98 }
99
100 /*
101 Note: at this stage the relative heights of the corners and the
102 centre are in the h array, and the corresponding coordinates are
103 in the xh and yh arrays. The centre of the box is indexed by 0
104 and the 4 corners by 1 to 4 as shown below.
105 Each triangle is then indexed by the parameter m, and the 3
106 vertices of each triangle are indexed by parameters m1,m2,and m3.
107 It is assumed that the centre of the box is always vertex 2
108 though this is important only when all 3 vertices lie exactly on
109 the same contour level, in which case only the side of the box
110 is drawn.
111 vertex 4 +-------------------+ vertex 3
112 | \ / |
113 | \ m=3 / |
114 | \ / |
115 | \ / |
116 | m=2 X m=2 | the centre is vertex 0
117 | / \ |
118 | / \ |
119 | / m=1 \ |
120 | / \ |
121 vertex 1 +-------------------+ vertex 2
122 */
123 /* Scan each triangle in the box */
124 for( size_t m=1; m <= 4; m++ ) {
125 size_t m1 = m;
126 size_t m2 = 0, m3;
127 if( m != 4 )
128 m3 = m + 1;
129 else
130 m3 = 1;
131 size_t case_value = castab[sh[m1]+1][sh[m2]+1][sh[m3]+1];
132 if( case_value == 0 )
133 continue;
134 Float x1=0, x2=0, y1=0, y2=0;
135 switch( case_value ) {
136 case 1: /* Line between vertices 1 and 2 */
137 x1 = xh[m1]; y1 = yh[m1];
138 x2 = xh[m2]; y2 = yh[m2];
139 break;
140 case 2: /* Line between vertices 2 and 3 */
141 x1 = xh[m2]; y1 = yh[m2];
142 x2 = xh[m3]; y2 = yh[m3];
143 break;
144 case 3: /* Line between vertices 3 and 1 */
145 x1 = xh[m3]; y1 = yh[m3];
146 x2 = xh[m1]; y2 = yh[m1];
147 break;
148 case 4: /* Line between vertex 1 and side 2-3 */
149 x1 = xh[m1]; y1 = yh[m1];
150 x2 = xsect(m2,m3); y2 = ysect(m2,m3);
151 break;
152 case 5: /* Line between vertex 2 and side 3-1 */
153 x1 = xh[m2]; y1 = yh[m2];
154 x2 = xsect(m3,m1); y2 = ysect(m3,m1);
155 break;
156 case 6: /* Line between vertex 3 and side 1-2 */
157 x1 = xh[m3]; y1 = yh[m3];
158 x2 = xsect(m3,m2); y2 = ysect(m3,m2);
159 break;
160 case 7: /* Line between sides 1-2 and 2-3 */
161 x1 = xsect(m1,m2); y1 = ysect(m1,m2);
162 x2 = xsect(m2,m3); y2 = ysect(m2,m3);
163 break;
164 case 8: /* Line between sides 2-3 and 3-1 */
165 x1 = xsect(m2,m3); y1 = ysect(m2,m3);
166 x2 = xsect(m3,m1); y2 = ysect(m3,m1);
167 break;
168 case 9: /* Line between sides 3-1 and 1-2 */
169 x1 = xsect(m3,m1); y1 = ysect(m3,m1);
170 x2 = xsect(m1,m2); y2 = ysect(m1,m2);
171 break;
172 default:
173 break;
174 }
175 /* Finally draw the line */
176 callback.OnLine(x1, y1, x2, y2, z[k]);
177 } /* m */
178 } /* k - contour */
179 } /* i */
180 } /* j */
181 }
182};
